114LoggerBuffer::int_type LoggerBuffer::overflow(int_type character)
115{
116 if (character != traits_type::eof()) {
117 // Write the charater to the buffer and then increment pptr() by
118 // calling pbump(1). It's always safe to write thecharater to the
119 // buffer because we reserved an extra char at the end of our buffer
120 // in the constructor.
121 assert(std::less_equal<char *>()(pptr(), epptr()));
122 *pptr() = character;
123 pbump(1);
124
125 // Flush the stream
126 if (sync() == 0) {
127 return character;
128 }
129 }
130
131 return traits_type::eof();
132}

151int LoggerBuffer::flush(bool terminate)
152{
153 int status = 0;
154 if (pptr() == pbase()) {
155 return status;
156 }
157
158 // Identify line breakers
159 std::vector<char *> linePointers;
160 linePointers.push_back(pbase());
161 for (char *p = pbase(), *e = pptr() - 1; p != e; ++p) {
162 if (*p == '\n') {
163 linePointers.push_back(p + 1);
164 }
165 }
166 linePointers.push_back(pptr());
167
168 // Detect if a new line must be added at the end
169 terminate = (terminate && *(linePointers.back() - 1) != '\n');
170
171 // Move the internal pointer
172 std::ptrdiff_t nCharacters = pptr() - pbase();
173 pbump(- nCharacters);
174
175 // Write all lines
176 for (unsigned int i = 0; i < linePointers.size() - 1; ++i) {
177 char *firstCharacter = linePointers[i];
178 char *lastCharacter = linePointers[i + 1];
179 std::ptrdiff_t lineSize = lastCharacter - firstCharacter;
180
181 // Write to the console
182 if (m_console && m_consoleEnabled) {
183 if (m_consoleTimestampEnabled) {
184 *m_console << "[" + getTimestamp() + "] ";
185 }
186
187 if (!m_consolePrefix.empty()) {
188 *m_console << m_consolePrefix << " :: ";
189 }
190
191 if (!m_context.empty()) {
192 *m_console << m_context << " :: ";
193 }
194
195 if (!m_padding.empty()) {
196 *m_console << m_padding;
197 }
198
199 m_console->write(firstCharacter, lineSize);
200 if ((m_console->rdstate() & std::ifstream::failbit ) != 0) {
201 status = -1;
202 }
203
204 if (terminate && lastCharacter == linePointers.back()) {
205 *m_console << "\n";
206 }
207 }
208
209 // Write to file
210 if (m_file && m_fileEnabled && m_file->is_open()) {
211 if (m_fileTimestampEnabled) {
212 *m_file << "[" + getTimestamp() + "] ";
213 }
214
215 if (!m_filePrefix.empty()) {
216 *m_file << m_filePrefix << " :: ";
217 }
218
219 if (!m_context.empty()) {
220 *m_file << m_context + " :: ";
221 }
222
223 if (!m_padding.empty()) {
224 *m_file << m_padding;
225 }
226
227 m_file->write(firstCharacter, lineSize);
228 if ((m_file->rdstate() & std::ifstream::failbit ) != 0) {
229 status = -1;
230 }
231
232 if (terminate && lastCharacter == linePointers.back()) {
233 *m_file << "\n";
234 }
235 }
236 }
237
238 return status;
239}

436const std::string LoggerBuffer::getTimestamp() const
437{
438 auto currentClock = std::chrono::system_clock::now();
439 std::time_t currentTime = std::chrono::system_clock::to_time_t(currentClock);
440
441 auto seconds = std::chrono::duration_cast<std::chrono::seconds>(currentClock.time_since_epoch());
442 auto millisecs = std::chrono::duration_cast<std::chrono::milliseconds>(currentClock.time_since_epoch() - seconds);
443 std::ostringstream millisecsConverter;
444 millisecsConverter << (millisecs.count());
445
446 char buffer[80];
447 strftime(buffer, 80, "%Y-%m-%d %H:%M:%S", localtime(&currentTime));
448 std::string timestamp(buffer);
449 timestamp += "." + millisecsConverter.str();
450
451 return timestamp;
452}

469Logger::Logger(const std::string &name,
470 std::ostream *consoleStream, std::ofstream *fileStream,
471 int nProcessors, int rank)

493 m_buffer.setConsolePrefix("");
494 m_buffer.setFilePrefix("");

499 setFileEnabled(m_defaultSeverity, m_defaultVisibility);

1198Logger & LoggerManager::cout(const std::string &name, log::Level defaultSeverity, log::Visibility defaultVisibility)
1199{
1200 // Get the logger
1201 if (m_loggers.count(name) == 0) {
1202 if (!isInitialized()) {
1203 setMode(log::MODE_SEPARATE);
1204 }
1205
1206 if (name == m_defaultName) {
1207 _create(name, false, m_defaultDirectory, 1, 0);
1208 } else {
1209 create(name, false, m_defaultDirectory, 1, 0);
1210 }
1211 }
1212
1213 Logger &logger = *(m_loggers.at(name));
1214
1215 // Set default properties
1216 if (defaultSeverity != log::Level::NOTSET) {
1217 logger.setDefaultSeverity(defaultSeverity);
1218 }
1219
1220 if (defaultVisibility != log::VISIBILITY_NOTSET) {
1221 logger.setDefaultVisibility(defaultVisibility);
1222 }
1223
1224 // Return the logger
1225 return logger;
1226}

1500void LoggerManager::create(const std::string &name, bool reset,
1501 const std::string &directory,
1502 int nProcessors, int rank)
1503{
1504 // Its not possible to create a log with the default name nor a log
1505 // with the same name of an existent logger nor a log with an empty
1506 // name
1507 if (exists(name)) {
1508 m_loggerUsers[name]++;
1509
1510 cout(name) << "Detected an attemp to create the logger \"" << name << "\" twice" << std::endl;
1511 cout(name) << "The previously created logger will be used." << std::endl;
1512 return;
1513 } else if (name == m_defaultName) {
1514 cout(m_defaultName) << "Detected an attemp to overwrite the default logger" << std::endl;
1515 cout(m_defaultName) << "The name of the default logger is \"" << name << "\"" << std::endl;
1516 return;
1517 } else if (name == "") {
1518 cout() << "Detected an attemp to create a logger with an empty name" << std::endl;
1519 return;
1520 }
1521
1522 // Create the logger
1523 if (m_mode == log::MODE_SEPARATE) {

1541bool LoggerManager::destroy(const std::string &name, bool force)
1542{
1543 if (!exists(name)) {
1544 return false;
1545 }
1546
1547 // Decrement the users
1548 int &nUsers = m_loggerUsers[name];
1549 nUsers--;
1550
1551 // If the logger has no users we can delete it
1552 if (nUsers == 0 || force) {
1553 // Remove the logger from the manager
1554 m_loggers.erase(name);
1555 m_loggerUsers.erase(name);
1556
1557 // If the logger has it own file stream, delete it
1558 if (m_fileStreams.count(name) > 0) {
1559 // Close the stream
1560 std::ofstream &fileStream = *(m_fileStreams.at(name));
1561 fileStream.close();
1562
1563 // Remove the stream
1564 m_fileStreams.erase(name);
1565 }
1566
1567 return true;
1568 } else {
1569 return false;
1570 }

1643void LoggerManager::_create(const std::string &name, bool reset,
1644 const std::string &directory,
1645 int nProcessors, int rank)
1646{
1647 // Get the file path
1648 FileHandler fileHandler;
1649 fileHandler.setDirectory(directory);
1650 fileHandler.setName(name);
1651 fileHandler.setAppendix("log");
1652 fileHandler.setParallel(nProcessors > 1);
1653 if (nProcessors > 1) {
1654 fileHandler.setBlock(rank);
1655 }
1656
1657 std::string filePath = fileHandler.getPath();
1658
1659 // Create the file stream
1660 std::ios_base::openmode fileMode;
1661 if (reset) {
1662 fileMode = std::ofstream::out;
1663 } else {
1664 fileMode = std::ofstream::app;
1665 }
1666
1667 m_fileStreams[name] = std::unique_ptr<std::ofstream>(new std::ofstream());
1668 std::ofstream &fileStream = *(m_fileStreams.at(name));
1669 fileStream.rdbuf()->pubsetbuf(0, 0);
1670 fileStream.open(filePath, fileMode);
1671
1672 // Use cout as console stream
1673 std::ostream &consoleStream = std::cout;
1674
1675 // Create the logger
1676 m_loggers[name] = std::unique_ptr<Logger>(new Logger(name, &consoleStream, &fileStream, nProcessors, rank));
1677 m_loggerUsers[name] = 1;
1678}
