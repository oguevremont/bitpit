149struct DoubleFloatingEqual : public DoubleFloatingComparison
150{
163 bool operator()(double x, double y, double relativeTolerance = DEFAULT_REL_TOL, double absoluteTolerance = DEFAULT_ABS_TOL) const
164 {
165 // Check if the numbers are really close (needed when comparing
166 // numbers near zero).
167 double absoluteDifference = std::abs(x - y);
168 if (absoluteDifference <= absoluteTolerance) {
169 return true;
170 }
171
172 // Compare using a relative difference
173 //
174 // The check that is performed is:
175 //
176 // abs(x - y) <= abs(x) * epsilon OR abs(x - y) <= abs(y) * epsilon
177 //
178 // The multiplication in the right side of inequalities may cause an
179 // unwanted underflow condition. To prevent this, the difference is
180 // scaled rather than epsilon.
181 double relativeDifference = absoluteDifference / std::max(std::abs(x), std::abs(y));
182
183 return (relativeDifference <= relativeTolerance);
184 }
185
186};
