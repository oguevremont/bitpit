362template<typename stencil_t>
363void DiscretizationStencilSolverAssembler<stencil_t>::setMatrixSizes(long nRows, long nCols)
364#endif
365{
366 // Set system sizes
367 m_nRows = nRows;
368 m_nCols = nCols;
369
370#if BITPIT_ENABLE_MPI==1
371 // Global system sizes
372 m_nGlobalRows = nRows;
373 m_nGlobalCols = nCols;
374 if (partitioned) {
375 MPI_Allreduce(MPI_IN_PLACE, &m_nGlobalRows, 1, MPI_LONG, MPI_SUM, communicator);
376 MPI_Allreduce(MPI_IN_PLACE, &m_nGlobalCols, 1, MPI_LONG, MPI_SUM, communicator);
377 }
378
379 // Global offsets
380 m_globalRowOffset = 0;
381 m_globalColOffset = 0;
382 if (partitioned) {
383 int nProcessors;
384 MPI_Comm_size(communicator, &nProcessors);
385
386 std::vector<long> nRankRows(nProcessors);
387 MPI_Allgather(&m_nRows, 1, MPI_LONG, nRankRows.data(), 1, MPI_LONG, communicator);
388
389 std::vector<long> nRankCols(nProcessors);
390 MPI_Allgather(&m_nCols, 1, MPI_LONG, nRankCols.data(), 1, MPI_LONG, communicator);
391
392 int rank;
393 MPI_Comm_rank(communicator, &rank);
394 for (int i = 0; i < rank; ++i) {
395 m_globalRowOffset += nRankRows[i];
396 m_globalColOffset += nRankCols[i];
397 }
398 }
399#endif
400}

554template<typename stencil_t>
555void DiscretizationStencilSolverAssembler<stencil_t>::getPattern(const stencil_t &stencil, ConstProxyVector<long> *pattern) const
556{
557 std::size_t stencilSize = stencil.size();
558
559 const long *patternData = stencil.patternData();
560 if (m_blockSize == 1) {
561 pattern->set(patternData, stencilSize);
562 } else {
563 std::size_t expandedPatternSize = m_blockSize * stencilSize;
564 pattern->set(ConstProxyVector<long>::INTERNAL_STORAGE, expandedPatternSize);
565 ConstProxyVector<long>::storage_pointer expandedPatternStorage = pattern->storedData();
566
567 std::size_t expandedPatternIdx = 0;
568 for (std::size_t k = 0; k < stencilSize; ++k) {
569 long patternBlockOffset = patternData[k] * m_blockSize;
570 for (int i = 0; i < m_blockSize; ++i) {
571 expandedPatternStorage[expandedPatternIdx++] = patternBlockOffset + i;
572 }
573 }
574 }

968{

972#else
973 DiscretizationStencilSolverAssembler<stencil_t> assembler(&stencils);
974#endif

978 for (std::size_t n = 0; n < nRows; ++n) {
979 long row;
980 if (rows) {
981 row = rows[n];
982 } else {
983 row = n;
984 }
985
986 m_constants[row] = assembler.getRowConstant(n);

993template<typename stencil_t>
994void DiscretizationStencilSolver<stencil_t>::solve()
995{
996 // Check if the stencil solver is assembled
997 if (!isAssembled()) {
998 throw std::runtime_error("Unable to solve the system. The stencil solver is not yet assembled.");
999 }
1000
1001 // Subtract constant terms to the RHS
1002 long nUnknowns = getRowCount();
1003 double *raw_rhs = getRHSRawPtr();
1004 for (long i = 0; i < nUnknowns; ++i) {
1005 raw_rhs[i] -= m_constants[i];
1006 }
1007 restoreRHSRawPtr(raw_rhs);
1008
1009 // Solve the system
1010 SystemSolver::solve();
1011}
