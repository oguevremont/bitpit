78template<typename kernel_t, template<typename> class narrow_band_cache_t>
79std::unique_ptr<LevelSetObject> LevelSetObjectFactory::_createImmutableObject(const kernel_t *kernel, LevelSetStorageType storageType, LevelSetObject *source)
80{
81 BITPIT_UNUSED(kernel);
82
83 switch (storageType) {
84
85 case LevelSetStorageType::SPARSE:
86 if (LevelSetBooleanObject *booleanSource = dynamic_cast<LevelSetBooleanObject *>(source)) {
87 return std::unique_ptr<LevelSetObject>(new LevelSetImmutableObject<narrow_band_cache_t<typename kernel_t::SparseStorageManager>>(booleanSource));
88 } else if (LevelSetCachedObject<narrow_band_cache_t<typename kernel_t::SparseStorageManager>> *cachedSource = dynamic_cast<LevelSetCachedObject<narrow_band_cache_t<typename kernel_t::SparseStorageManager>> *>(source)) {
89 return std::unique_ptr<LevelSetObject>(new LevelSetImmutableObject<narrow_band_cache_t<typename kernel_t::SparseStorageManager>>(cachedSource));
90 }
91 break;
92
93 case LevelSetStorageType::DENSE:
94 if (LevelSetBooleanObject *booleanSource = dynamic_cast<LevelSetBooleanObject *>(source)) {
95 return std::unique_ptr<LevelSetObject>(new LevelSetImmutableObject<narrow_band_cache_t<typename kernel_t::DenseStorageManager>>(booleanSource));
96 } else if (LevelSetCachedObject<narrow_band_cache_t<typename kernel_t::DenseStorageManager>> *cachedSource = dynamic_cast<LevelSetCachedObject<narrow_band_cache_t<typename kernel_t::DenseStorageManager>> *>(source)) {
97 return std::unique_ptr<LevelSetObject>(new LevelSetImmutableObject<narrow_band_cache_t<typename kernel_t::DenseStorageManager>>(cachedSource));
98 }
99 break;
100
101 default:
102 BITPIT_UNREACHABLE("Storage type not supported");
103
104 }
105
106 BITPIT_UNREACHABLE("Unable to create an immutable object from the specified object.");
107}
