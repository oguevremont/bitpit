291void Vertex::scale(const std::array<double, 3> &scaling, const std::array<double, 3> &center)
292{
293 m_coords[0] = center[0] + scaling[0] * (m_coords[0] - center[0]);
294 m_coords[1] = center[1] + scaling[1] * (m_coords[1] - center[1]);
295 m_coords[2] = center[2] + scaling[2] * (m_coords[2] - center[2]);
296}
