71template<typename id_t>
72PiercedStorageSyncSlave<id_t>::PiercedStorageSyncSlave(const PiercedStorageSyncSlave<id_t> &other, const PiercedKernel<id_t> *kernel)
73 : PiercedStorageSyncSlave<id_t>()
74{
75 KernelType kernelType = other.getKernelType();
76 switch (kernelType) {
77
78 case KERNEL_STATIC:
79 {
80 if (kernel) {
81 setStaticKernel(kernel);
82 } else {
83 setStaticKernel(other.m_const_kernel);
84 }
85 break;
86 }
87
88 case KERNEL_DYNAMIC:
89 {
90 throw std::runtime_error("Unable to set a dynamic kernel. The kernel received in input can be only used to set a static kernel");
91 }
92
93 default:
94 {
95 break;
96 }
97
98 }
99}

250template<typename id_t>
251void PiercedStorageSyncSlave<id_t>::setStaticKernel(const PiercedKernel<id_t> *kernel)
252{
253 // Set the kernel
254 if (!kernel) {
255 throw std::runtime_error("Unable to set the kernel. Provided kernel is not valid.");
256 } else if (m_kernel != nullptr) {
257 throw std::runtime_error("Unable to set the kernel. The kernel of the storage is already set.");
258 }
259
260 m_kernel = nullptr;
261 m_const_kernel = kernel;
262
263 // Action to be performed after setting the kernel
264 _postSetStaticKernel();
265}

349template<typename id_t>
350void PiercedStorageSyncSlave<id_t>::detachKernel()
351{
352 if (!m_const_kernel) {
353 return;
354 }
355

358 m_kernel = nullptr;
359 }

703template<typename value_t, typename id_t>
704void PiercedStorage<value_t, id_t>::commitSyncAction(const PiercedSyncAction &action)
705{
706 switch (action.type) {
707
708 case PiercedSyncAction::TYPE_CLEAR:
709 {
710 rawClear(true);
711 break;
712 }
713
714 case PiercedSyncAction::TYPE_RESERVE:
715 {
716 rawReserve(action.info[PiercedSyncAction::INFO_SIZE]);
717 break;
718 }
719
720 case PiercedSyncAction::TYPE_RESIZE:
721 {
722 rawResize(action.info[PiercedSyncAction::INFO_SIZE]);
723 break;
724 }
725
726 case PiercedSyncAction::TYPE_SHRINK_TO_FIT:
727 {
728 rawShrinkToFit();
729 break;
730 }
731
732 case PiercedSyncAction::TYPE_REORDER:
733 {
734 if (action.data && action.data->size() > 0) {
735 rawReorder(*action.data);
736 }
737 rawResize(action.info[PiercedSyncAction::INFO_SIZE]);
738 rawShrinkToFit();
739 break;
740 }
741
742 case PiercedSyncAction::TYPE_APPEND:
743 {
744 // Since we may increase the sotrage by an element at the time calling
745 // a reserve will hurt performance badly because this will prevent the
746 // automatic reallocation of the storage.
747 rawEmplaceBack();
748 break;
749 }
750
751 case PiercedSyncAction::TYPE_INSERT:
752 {
753 // Since we may increase the sotrage by an element at the time calling
754 // a reserve will hurt performance badly because this will prevent the
755 // automatic reallocation of the storage.
756 rawEmplace(action.info[PiercedSyncAction::INFO_POS]);
757 break;
758 }
759
760 case PiercedSyncAction::TYPE_OVERWRITE:
761 {
762 break;
763 }
764
765 case PiercedSyncAction::TYPE_MOVE_APPEND:
766 {
767 // Since we may increase the sotrage by an element at the time calling
768 // a reserve will hurt performance badly because this will prevent the
769 // automatic reallocation of the storage.
770 rawPushBack(std::move(rawAt(action.info[PiercedSyncAction::INFO_POS_FIRST])));
771 rawEmreplace(action.info[PiercedSyncAction::INFO_POS_FIRST]);
772 break;
773 }
774
775 case PiercedSyncAction::TYPE_MOVE_INSERT:
776 {
777 // Since we may increase the sotrage by an element at the time calling
778 // a reserve will hurt performance badly because this will prevent the
779 // automatic reallocation of the storage.
780 rawInsert(action.info[PiercedSyncAction::INFO_POS_SECOND], 1, std::move(rawAt(action.info[PiercedSyncAction::INFO_POS_FIRST])));
781 rawEmreplace(action.info[PiercedSyncAction::INFO_POS_FIRST]);
782 break;
783 }
784
785 case PiercedSyncAction::TYPE_MOVE_OVERWRITE:
786 {
787 rawSet(action.info[PiercedSyncAction::INFO_POS_SECOND], std::move(rawAt(action.info[PiercedSyncAction::INFO_POS_FIRST])));
788 rawEmreplace(action.info[PiercedSyncAction::INFO_POS_FIRST]);
789 break;
790 }
791
792 case PiercedSyncAction::TYPE_PIERCE:
793 case PiercedSyncAction::TYPE_PIERCE_MULTIPLE:
794 {
795 // Nothing to do. To improve performance the element will not be
796 // cleared.
797 break;
798 }
799
800 case PiercedSyncAction::TYPE_SWAP:
801 {
802 rawSwap(action.info[PiercedSyncAction::INFO_POS_FIRST], action.info[PiercedSyncAction::INFO_POS_SECOND]);
803 break;
804 }
805
806 case PiercedSyncAction::TYPE_NOOP:
807 {
808 break;
809 }
810
811 default:
812 {
813 throw std::runtime_error("Undefined synchronization action.");
814 }
815
816 }
817}

897template<typename value_t, typename id_t>
898void PiercedStorage<value_t, id_t>::rawReorder(const std::vector<std::size_t> &permutations)
899{
900 std::size_t storageRawSize = rawSize();
901 assert(permutations.size() == storageRawSize);
902
903 // Evaluate the permutation of the fields
904 std::vector<std::size_t> fieldPermutations(storageRawSize * m_nFields);
905 for (std::size_t pos = 0; pos < storageRawSize; ++pos) {
906 std::size_t posOffset = pos * m_nFields;
907 std::size_t permutationOffset = permutations[pos] * m_nFields;
908 for (std::size_t k = 0; k < m_nFields; ++k) {
909 fieldPermutations[posOffset + k] = permutationOffset + k;
910 }
911 }
912
913 // Sort the fields
914 utils::reorderVector<value_t>(fieldPermutations, m_fields, storageRawSize * m_nFields);
915}

1076template<typename value_t, typename id_t>
1077void PiercedStorage<value_t, id_t>::swap(PiercedStorage &other) noexcept
1078{
1079 // It is only possible to swap two storages with the same number of field.
1080 // If this condition is not fulfilled we can not continue. However, we
1081 // cannot throw an exception because the function is declared snoexcept.
1082 if (other.getFieldCount() != getFieldCount()) {
1083 std::cout << "It is only possible to swap storages with the same number of fields." << std::endl;
1084 assert(false);
1085 exit(EXIT_FAILURE);
1086 }
1087
1088 PiercedStorageSyncSlave<id_t>::swap(other);
1089 std::swap(other.m_nFields, m_nFields);
1090 std::swap(other.m_fields, m_fields);
1091}
