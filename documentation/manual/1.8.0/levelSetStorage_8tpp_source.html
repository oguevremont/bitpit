780template<typename kernel_t, typename kernel_iterator_t>
781void LevelSetStorageManager<kernel_t, kernel_iterator_t>::write(const std::vector<long> &ids, SendBuffer &buffer)
782{
783 // Evaluate the size of the buffer
784 std::size_t nBufferItems = 0;
785 for (long id : ids) {
786 if (contains(id)) {
787 ++nBufferItems;
788 }
789 }
790
791 std::size_t bufferSize = buffer.getSize() + sizeof(std::size_t);
792 for (const auto &storage : m_storages) {
793 bufferSize += nBufferItems * storage.second->getItemBinarySize();
794 }
795 buffer.setSize(bufferSize);
796
797 // Fill the buffer
798 buffer << nBufferItems;
799 for (std::size_t k = 0; k < ids.size(); ++k) {
800 // Get an iterator pointing to the item
801 long id = ids[k];
802 KernelIterator itr = find(id);
803 if (itr == end()) {
804 continue;
805 }
806
807 // Write the index of the cell
808 buffer << k;
809
810 // Write item data
811 for (const auto &storage : m_storages) {
812 storage.second->writeItem(itr, buffer);
813 }
814 }
815}

826template<typename kernel_t, typename kernel_iterator_t>
827void LevelSetStorageManager<kernel_t, kernel_iterator_t>::read(const std::vector<long> &ids, RecvBuffer &buffer)
828{
829 std::size_t nReceviedItems;
830 buffer >> nReceviedItems;
831
832 for (std::size_t i = 0; i < nReceviedItems; ++i) {
833 // Read the id of the item
834 std::size_t k;
835 buffer >> k;
836 long id = ids[k];
837
838 // Create the item
839 KernelIterator itr = find(id);
840 if (itr == end()) {
841 itr = insert(id);
842 }
843
844 // Read item data
845 for (auto &storage : m_storages) {
846 storage.second->readItem(itr, buffer);
847 }
848 }
849}
