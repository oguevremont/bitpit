256template<typename narrow_band_cache_t>
261double LevelSetSegmentationObject<narrow_band_cache_t>::getMinSurfaceFeatureSize( ) const {
262
263 const SurfUnstructured &m_surface = m_segmentation->getSurface();
264
265 bool minimumValid = false;
266 double minimumSize = levelSetDefaults::SIZE;
267 for( const Cell &cell : m_surface.getCells() ){
268 double segmentSize = getSegmentSize(cell.getId());
269 if (segmentSize < 0) {
270 continue;
271 }
272
273 minimumValid = true;
274 minimumSize = std::min(segmentSize, minimumSize);
275 }
276
277 if (!minimumValid) {
278 minimumSize = - levelSetDefaults::SIZE;
279 }
280
281 return minimumSize;
282}

391template<typename narrow_band_cache_t>
392void LevelSetSegmentationObject<narrow_band_cache_t>::computeNarrowBand( LevelSetCartesianKernel *levelsetKernel, bool signd){
393
394 log::cout() << " Compute levelset on cartesian mesh" << std::endl;
395
396 // Get mesh information
397 const VolCartesian &mesh = *(levelsetKernel->getCartesianMesh() ) ;
398 int meshDimension = mesh.getDimension();
399 VolCartesian::MemoryMode meshMemoryMode = mesh.getMemoryMode();
400
401 ElementType meshCellType = mesh.getCellType();
402 const ReferenceElementInfo &meshCellTypeInfo = ReferenceElementInfo::getInfo(meshCellType);
403 int meshCellFaceCount = meshCellTypeInfo.nFaces;
404
405 // Get surface information
406 const SurfUnstructured &surface = m_segmentation->getSurface();
407
408 // Define search radius
409 //
410 // Search radius should be equal to the maximum between the narrow band
411 // size and the diameter of the circumcircle. This guarantees that, when
412 // the narrow band size is equal or less than zero, the levelset will be
413 // evaluated on the cells that intersect the surface and on all their
414 // first neighbours.
415 double searchRadius = std::max(this->m_narrowBandSize, 2 * levelsetKernel->getCellCircumcircle());
416
417 // Define mesh bounding box
418 //
419 // The bounding box is inflated be the search radius.
420 std::array<double,3> meshMinPoint;
421 std::array<double,3> meshMaxPoint;
422 mesh.getBoundingBox(meshMinPoint, meshMaxPoint) ;
423 for (int d = 0; d < meshDimension; ++d) {
424 meshMinPoint[d] -= searchRadius;
425 meshMaxPoint[d] += searchRadius;
426 }
427
428 // Initialize process list
429 //
430 // Process list is initialized with cells that are certainly inside the
431 // narrow band. Those cells are the one that contain the vertices of the
432 // segments or the intersection between the segments and the bounding box
433 // of the patch.
434 std::unordered_set<long> processList;
435
436 std::vector<std::array<double,3>> intersectionPoints;
437 std::vector<std::array<double,3>> segmentVertexCoords;
438 for (const Cell &segment : surface.getCells()) {
439 // Get segment info
440 ConstProxyVector<long> segmentVertexIds = segment.getVertexIds();
441 std::size_t nSegmentVertices = segmentVertexIds.size();
442
443 // Get segment coordinates
444 segmentVertexCoords.resize(nSegmentVertices);
445 surface.getVertexCoords(nSegmentVertices, segmentVertexIds.data(), segmentVertexCoords.data());
446
447 // Add to the process list the cells that contain the vertices of the
448 // segment or the intersection between the segment and the bounding box
449 // of the patch.
450 int nInnerVertices = 0;
451 for (const std::array<double,3> &vertexPoint : segmentVertexCoords) {
452 long cellId = mesh.locatePoint(vertexPoint);
453 if (cellId < 0) {
454 continue;
455 }
456
457 processList.insert(cellId);
458 ++nInnerVertices;
459 }
460
461 if (nInnerVertices == 0) {
462 if (CGElem::intersectBoxPolygon(meshMinPoint, meshMaxPoint, segmentVertexCoords, false, true, true, intersectionPoints, meshDimension)) {
463 for (const std::array<double,3> &intersectionPoint : intersectionPoints){
464 long cellId = mesh.locateClosestCell(intersectionPoint);
465 processList.insert(cellId);
466 }
467 }
468 }
469 }
470
471 // Evaluate the levelset within the narrow band
472 //
473 // The initial process list is gradually expanded considering all the
474 // neighbours with a distance less than the search radius.
475 narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
476
477 std::unordered_set<long> outsideNarrowBand;
478 while (!processList.empty()) {
479 // Get the cell to process
480 long cellId = *(processList.begin());
481 processList.erase(cellId);
482
483 // Find segment associated to the cell
484 const std::array<double,3> &cellCentroid = levelsetKernel->computeCellCentroid(cellId);
485
486 long segmentId;
487 double distance;
488 m_segmentation->getSearchTree().findPointClosestCell(cellCentroid, searchRadius, &segmentId, &distance);
489 if(segmentId < 0){
490 outsideNarrowBand.insert(cellId);
491 continue;
492 }
493
494 // Evaluate levelset information
495 std::array<double, 3> gradient;
496 std::array<double, 3> normal;
497 int error = m_segmentation->getSegmentInfo(cellCentroid, segmentId, signd, distance, gradient, normal);
498 if (error) {
499 throw std::runtime_error ("Unable to extract the levelset information from segment.");
500 }
501
502
503 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(cellId, true) ;
504 narrowBandCache->set(narrowBandCacheItr, distance, gradient, segmentId, normal);
505
506 // Add cell neighbours to the process list
507 if (meshMemoryMode == VolCartesian::MEMORY_LIGHT) {
508 for (int face = 0; face < meshCellFaceCount; ++face) {
509 long neighId = mesh.getCellFaceNeighsLinearId(cellId, face);
510 if (neighId >= 0) {
511 if (!this->isInNarrowBand(neighId) && (outsideNarrowBand.count(neighId) == 0)) {
512 processList.insert(neighId);
513 }
514 }
515 }
516 } else {
517 const Cell &cell = mesh.getCell(cellId);
518 const long *neighbours = cell.getAdjacencies() ;
519 int nNeighbours = cell.getAdjacencyCount() ;
520 for (int n = 0; n < nNeighbours; ++n) {
521 long neighId = neighbours[n];
522 if (!this->isInNarrowBand(neighId) && (outsideNarrowBand.count(neighId) == 0)) {
523 processList.insert(neighId);
524 }
525 }
526 }
527 }
528}

542template<typename narrow_band_cache_t>
543void LevelSetSegmentationObject<narrow_band_cache_t>::computeNarrowBand( LevelSetOctreeKernel *levelsetKernel, bool signd){
544
545 VolumeKernel &mesh = *(levelsetKernel->getMesh()) ;
546
547 std::unordered_set<long> intersectedCells;
548
549 // Evaluate levelset information
550 narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
551
552 for( const Cell &cell : mesh.getCells() ){
553
554 // Identify the segment associated with the cell
555 //
556 // The search radius is evaluated as the maximum value between the
557 // narroband size and the distance above which the cell will surely
558 // not intersect the surface. In this way, cells that intersect the
559 // surface are always included in the narrowband, even if their
560 // distance from the surface is greater than then narrowband size
561 // explicitly set by the user.
562 //
563 // If no segment is identified the cell is not processed.
564 long cellId = cell.getId();
565 const std::array<double,3> &cellCentroid = levelsetKernel->computeCellCentroid(cellId);
566 double cellCircumcircle = levelsetKernel->computeCellCircumcircle(cellId);
567
568 double searchRadius = std::max(this->m_narrowBandSize, cellCircumcircle);
569
570 long segmentId;
571 double distance;
572 m_segmentation->getSearchTree().findPointClosestCell(cellCentroid, searchRadius, &segmentId, &distance);
573 if(segmentId < 0){
574 continue;
575 }
576
577 // Evaluate levelset information
578 std::array<double,3> gradient;
579 std::array<double,3> normal;
580 int error = m_segmentation->getSegmentInfo(cellCentroid, segmentId, signd, distance, gradient, normal);
581 if (error) {
582 throw std::runtime_error ("Unable to extract the levelset information from segment.");
583 }
584
585 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(cellId, true) ;
586 narrowBandCache->set(narrowBandCacheItr, distance, gradient, segmentId, normal);
587
588 // Update the list of cells that intersects the surface
589 //
590 // When the narrowband size is not explicitly set, the cell will always
591 // intersects the surface because only cells that intersect the surface
592 // are considered, otherwise we need to check if the absolute distance
593 // associated with the cell is lower than the intersection distance.
594 if (this->m_narrowBandSize < 0 || cellCircumcircle < std::abs(distance)) {
595 intersectedCells.insert(cellId);
596 }
597
598 }
599
600 // Process the neighbours of the cells that intersect the surface
601 //
602 // If a cell intersects the surface, we need to evaluate the levelset
603 // of all its neigbours.
604 for( long cellId : intersectedCells){
605
606 Cell const &cell = mesh.getCell(cellId);
607
608 std::array<double,3> cellProjectionPoint = this->computeProjectionPoint(cellId);
609
610 const long *neighbours = cell.getAdjacencies() ;
611 int nNeighbours = cell.getAdjacencyCount() ;
612 for (int n = 0; n < nNeighbours; ++n) {
613 // Skip the neighbour if it has already been processed
614 //
615 // The neighbour may already have been processed either because
616 // it distance from the segmentation is within the search radius,
617 // or because is a neighbour of an intersected cells already
618 // processed.
619 long neighId = neighbours[n];
620 if( narrowBandCache->contains(neighId) ){
621 continue;
622 }
623
624 // Identify the segment associated with the neighbour
625 const std::array<double,3> &neighCentroid = levelsetKernel->computeCellCentroid(neighId);
626
627 double searchRadius = 1.05 * norm2(neighCentroid - cellProjectionPoint);
628
629 long segmentId;
630 double distance;
631 m_segmentation->getSearchTree().findPointClosestCell(neighCentroid, searchRadius, &segmentId, &distance);
632 if (segmentId < 0) {
633 assert(false && "Should not pass here");
634 }
635
636 // Evaluate negihbour leveset information
637 std::array<double,3> gradient;
638 std::array<double,3> normal;
639 int error = m_segmentation->getSegmentInfo(neighCentroid, segmentId, signd, distance, gradient, normal);
640 if (error) {
641 throw std::runtime_error ("Unable to extract the levelset information from segment.");
642 }
643
644 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(neighId, true) ;
645 narrowBandCache->set(narrowBandCacheItr, distance, gradient, segmentId, normal);
646 }
647 }
648}

664template<typename narrow_band_cache_t>
665void LevelSetSegmentationObject<narrow_band_cache_t>::updateNarrowBand( LevelSetOctreeKernel *levelsetKernel, const std::vector<adaption::Info> &adaptionData, bool signd){
666
667 VolumeKernel &mesh = *(levelsetKernel->getMesh()) ;
668 narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
669
670 std::vector<long> cellsOutsideNarrowband;
671
672 // Evaluate the levelset of the cells
673 //
674 // When searching for the segment associated to a cell, the search radius
675 // is evaluated as the maximum value between the narroband size and the
676 // distance above which the cell will surely not intersect the surface.
677 for( const adaption::Info &adaptionInfo : adaptionData ){
678
679 if( adaptionInfo.entity != adaption::Entity::ENTITY_CELL ){
680 continue;
681 }
682
683 if( adaptionInfo.type == adaption::Type::TYPE_PARTITION_SEND){
684 continue;
685 } else if( adaptionInfo.type == adaption::Type::TYPE_PARTITION_RECV){
686 continue;
687 }
688
689 for( long cellId : adaptionInfo.current ){
690
691 // Identify the segment associated with the cell
692 //
693 // The search radius is evaluated as the maximum value between the
694 // narroband size and the distance above which the cell will surely
695 // not intersect the surface. In this way, cells that intersect the
696 // surface are always included in the narrowband, even if their
697 // distance from the surface is greater than then narrowband size
698 // explicitly set by the user.
699 //
700 // If no segment is identified the cell is not processed.
701 const std::array<double,3> &centroid = levelsetKernel->computeCellCentroid(cellId);
702
703 double searchRadius = std::max(this->m_narrowBandSize, levelsetKernel->computeCellCircumcircle(cellId));
704
705 long segmentId;
706 double distance;
707 m_segmentation->getSearchTree().findPointClosestCell(centroid, searchRadius, &segmentId, &distance);
708 if (segmentId < 0) {
709 cellsOutsideNarrowband.push_back(cellId);
710 continue;
711 }
712
713 // Evaluate levelset information
714 std::array<double,3> gradient;
715 std::array<double,3> normal;
716 int error = m_segmentation->getSegmentInfo(centroid, segmentId, signd, distance, gradient, normal);
717 if (error) {
718 throw std::runtime_error ("Unable to extract the levelset information from segment.");
719 }
720
721 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(cellId, true) ;
722 narrowBandCache->set(narrowBandCacheItr, distance, gradient, segmentId, normal);
723 }
724
725 }
726
727 // Cells with neighbours that intersect the surface need to be added to
728 // the narrowband even if they don't intersect the surface themself or
729 // have a distance from the surface greater than the narroband size.
730 for( long cellId : cellsOutsideNarrowband){
731 const Cell &cell = mesh.getCell(cellId);
732
733 // Consider only cells with a neighbour that intersects the surface
734 //
735 // Care must be take to use only information from cells inside the
736 // narrow band, that's because values outside the narrowband are not
737 // up-to-date at this stage.
738 const long *neighbours = cell.getAdjacencies() ;
739 int nNeighbours = cell.getAdjacencyCount() ;
740
741 long intersectedNeighId = Cell::NULL_ID;
742 for (int n = 0; n < nNeighbours; ++n) {
743 long neighId = neighbours[n];
744 if (!this->isInNarrowBand(neighId)) {
745 continue;
746 }
747
748 if( this->intersectSurface(neighId,LevelSetIntersectionMode::FAST_GUARANTEE_FALSE) == LevelSetIntersectionStatus::TRUE){
749 intersectedNeighId = neighId;
750 break;
751 }
752 }
753
754 if (intersectedNeighId == Cell::NULL_ID) {
755 continue;
756 }
757
758 // Identify the segment associated with the cell
759 const std::array<double,3> &cellCentroid = levelsetKernel->computeCellCentroid(cellId);
760 std::array<double,3> neighProjectionPoint = this->computeProjectionPoint(intersectedNeighId);
761
762 double searchRadius = 1.05 * norm2(cellCentroid - neighProjectionPoint);
763
764 long segmentId;
765 double distance;
766 m_segmentation->getSearchTree().findPointClosestCell(cellCentroid, searchRadius, &segmentId, &distance);
767 if (segmentId < 0) {
768 assert(false && "Should not pass here");
769 continue;
770 }
771
772 // Evaluate levelset information for the cell
773 std::array<double,3> gradient;
774 std::array<double,3> normal;
775 int error = m_segmentation->getSegmentInfo(cellCentroid, segmentId, signd, distance, gradient, normal);
776 if (error) {
777 throw std::runtime_error ("Unable to extract the levelset information from segment.");
778 }
779
780 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(cellId, true) ;
781 narrowBandCache->set(narrowBandCacheItr, distance, gradient, segmentId, normal);
782 }
