103void SegmentationKernel::setSurface( const SurfUnstructured *surface, double featureAngle){
104
105 // Check if adjacencies are built
106 if (surface->getAdjacenciesBuildStrategy() == SurfUnstructured::ADJACENCIES_NONE) {
107 throw std::runtime_error ("Segmentation needs adjacencies!") ;
108 }
109
110 // Surface information
111 m_surface = surface;
112 m_featureAngle = featureAngle;
113
114 // Check if segment is supported
115 SurfUnstructured::CellConstIterator endItr = m_surface->cellConstEnd();
116 for( SurfUnstructured::CellConstIterator segmentItr = m_surface->cellConstBegin(); segmentItr != endItr; ++segmentItr ){
117 switch (segmentItr->getType()) {
118
119 case ElementType::VERTEX :
120 case ElementType::LINE :
121 case ElementType::TRIANGLE :
122 break ;
123
124 default:
125 throw std::runtime_error ("levelset: only segments and triangles supported in LevelSetSegmentation!") ;
126 break ;
127
128 }
129 }
130
131 // Segment vertices information
132 m_segmentVertexOffset.setStaticKernel(&m_surface->getCells());
133
134 std::size_t nTotalSegmentVertices = 0;
135 for( auto itr = m_segmentVertexOffset.begin(); itr != m_segmentVertexOffset.end(); ++itr ){
136 *itr = nTotalSegmentVertices;
137 nTotalSegmentVertices += m_surface->getCells().rawAt(itr.getRawIndex()).getVertexCount();
138 }
139
140 // Normals
141 m_segmentNormalsValid.setStaticKernel(&m_surface->getCells());
142 m_segmentNormalsValid.fill(false);
143 m_segmentNormalsStorage.setStaticKernel(&m_surface->getCells());
144
145 m_unlimitedVertexNormalsValid.setStaticKernel(&m_surface->getVertices());
146 m_unlimitedVertexNormalsValid.fill(false);
147 m_unlimitedVertexNormalsStorage.setStaticKernel(&m_surface->getVertices());
148
149 m_limitedSegmentVertexNormalValid.resize(nTotalSegmentVertices);
150
151 // Initialize search tree
152 m_searchTree = std::unique_ptr<SurfaceSkdTree>(new SurfaceSkdTree(surface));
153 m_searchTree->build();
154}

166int SegmentationKernel::getSegmentInfo( const std::array<double,3> &pointCoords, long segmentId, bool signd, double &distance, std::array<double,3> &gradient, std::array<double,3> &normal ) const {
167
168 // Segment information
169 SurfUnstructured::CellConstIterator segmentIterator = m_surface->getCellConstIterator(segmentId);
170 const Cell &segment = *segmentIterator ;
171 ElementType segmentType = segment.getType();
172 ConstProxyVector<long> segmentVertexIds = segment.getVertexIds() ;
173 int nSegmentVertices = segmentVertexIds.size() ;
174
175 // Projct the point on the surface and evaluate the point-projeciont vector
176 BITPIT_CREATE_WORKSPACE(lambda, double, nSegmentVertices, ReferenceElementInfo::MAX_ELEM_VERTICES);
177 std::array<double,3> pointProjectionVector = pointCoords;
178 switch (segmentType) {
179
180 case ElementType::VERTEX :
181 {
182 long id = segmentVertexIds[0] ;
183 pointProjectionVector -= m_surface->getVertexCoords(id);
184
185 break;
186 }
187
188 case ElementType::LINE:
189 {
190 long id0 = segmentVertexIds[0] ;
191 long id1 = segmentVertexIds[1] ;
192 pointProjectionVector -= CGElem::projectPointSegment( pointCoords, m_surface->getVertexCoords(id0), m_surface->getVertexCoords(id1), lambda);
193
194 break;
195 }
196
197 case ElementType::TRIANGLE:
198 {
199 long id0 = segmentVertexIds[0] ;
200 long id1 = segmentVertexIds[1] ;
201 long id2 = segmentVertexIds[2] ;
202 pointProjectionVector -= CGElem::projectPointTriangle( pointCoords, m_surface->getVertexCoords(id0), m_surface->getVertexCoords(id1), m_surface->getVertexCoords(id2), lambda );
203
204 break;
205 }
206
207 default:
208 {
209 std::runtime_error ("Type of cell not supported.");
210 break;
211 }
212
213 }
214
215 // Compute surface normal
216 normal = computeSurfaceNormal(segmentIterator, lambda);
217
218 // Evaluate distance from surface
219 distance = norm2(pointProjectionVector);
220
221 // Check if the point lies on the segmentation
222 //
223 // If the distance is zero, the point and the projection are coincident,
224 // this means that the point lies on the segmentation.
225 double distanceTolerance = m_surface->getTol();
226 bool pointOnSegmentation = utils::DoubleFloatingEqual()(distance, 0., distanceTolerance, distanceTolerance);
227
228 // Evaluate levelset gradient
229 if (!pointOnSegmentation) {
230 gradient = pointProjectionVector / distance;
231 } else {
232 if (signd) {
233 gradient = normal;
234 } else {
235 gradient = {{0., 0., 0.}};
236 }
237 }
238
239 // Evaluate levelset sign
240 //
241 // The sign is computed by determining the side of the point with respect
242 // to the normal plane. The sign will be zero if the point lies exaclty
243 // on the segmentation or on the normal plane. In the latter case the sign
244 // must be evaluated taking into account the the curvature of the surface.
245 // However, this is not yet implemented.
246 std::array<double, 3> pseudoNormal = computePseudoNormal(segmentIterator, lambda);
247 int s = sign( dotProduct(pointProjectionVector, pseudoNormal) );
248 if (!pointOnSegmentation && s == 0) {
249 distance = levelSetDefaults::VALUE;
250 gradient = levelSetDefaults::GRADIENT;
251 normal = levelSetDefaults::GRADIENT;
252
253 return 1;
254 }
255
256 // Use sign to update levelset information
257 //
258 // If signed distance are computed, the distance value and gradient
259 // need to be changed accordingly. If unsigned distance are computed
260 // the orientation of the suraface normal is discarded and in order
261 // to agnostic with repect the two sides of the surface
262 if (s < 0) {
263 distance *= (double) ( signd *s + (!signd) *1);
264 gradient *= (double) ( signd *s + (!signd) *1);
265 normal *= (double) ( signd *1 + (!signd) *s);
266 }
267
268 return 0;
269}

293std::array<double,3> SegmentationKernel::computePseudoNormal( const SurfUnstructured::CellConstIterator &segmentIterator, const double *lambda ) const {
294
295 // Early return if the segment is a point
296 const Cell &segment = *segmentIterator;
297 ElementType segmentType = segment.getType();
298 if (segmentType == ElementType::VERTEX) {
299 return {{0., 0., 0.}};
300 }
301
302 // Evaluate pseudo normal
303 int positionFlag;
304 if (segmentType == ElementType::LINE) {
305 positionFlag = CGElem::convertBarycentricToFlagSegment(lambda, m_surface->getTol());
306 } else {
307 int nSegmentVertices = segment.getVertexCount();
308 positionFlag = CGElem::convertBarycentricToFlagPolygon(nSegmentVertices, lambda, m_surface->getTol());
309 }
310
311 std::array<double,3> pseudoNormal;
312 if (positionFlag == 0) {
313 pseudoNormal = computeSegmentNormal(segmentIterator);
314 } else if (positionFlag > 0) {
315 int vertex = positionFlag - 1;
316 pseudoNormal = computeSegmentVertexNormal(segmentIterator, vertex, false);
317 } else if (positionFlag < 0) {
318 int edge = (- positionFlag) - 1;
319 pseudoNormal = computeSegmentEdgeNormal(segmentIterator, edge);
320 }
321
322 return pseudoNormal;
323}

337std::array<double,3> SegmentationKernel::computeSurfaceNormal( const SurfUnstructured::CellConstIterator &segmentIterator, const double *lambda ) const {
338
339 // Early return if the segment is a point
340 const Cell &segment = *segmentIterator;
341 ElementType segmentType = segment.getType();
342 if (segmentType == ElementType::VERTEX) {
343 return {{0., 0., 0.}};
344 }
345
346 // Evaluate surface normal
347 std::size_t nSegmentVertices = segment.getVertexCount();
348 std::array<double,3> surfaceNormal = lambda[0] * computeSegmentVertexNormal(segmentIterator, 0, true);
349 for (std::size_t i = 1; i < nSegmentVertices; ++i) {
350 surfaceNormal += lambda[i] * computeSegmentVertexNormal(segmentIterator, i, true);
351 }
352 surfaceNormal /= norm2(surfaceNormal);
353
354 return surfaceNormal;
355}

365std::array<double,3> SegmentationKernel::computeSegmentNormal( const SurfUnstructured::CellConstIterator &segmentIterator ) const {
366
367 std::size_t segmentRawId = segmentIterator.getRawIndex();
368 std::array<double, 3> *segmentNormal = m_segmentNormalsStorage.rawData(segmentRawId);
369 if (!m_segmentNormalsValid.rawAt(segmentRawId)) {
370 *segmentNormal = m_surface->evalFacetNormal(segmentIterator->getId());
371 m_segmentNormalsValid.rawAt(segmentRawId) = true;
372 }
373
374 return *segmentNormal;
375}

386std::array<double,3> SegmentationKernel::computeSegmentEdgeNormal( const SurfUnstructured::CellConstIterator &segmentIterator, int edge ) const {
387
388 std::array<double,3> normal = computeSegmentNormal(segmentIterator);
389
390 if (segmentIterator->getAdjacencyCount(edge) > 0) {
391 long neighId = segmentIterator->getAdjacency(edge);
392 SurfUnstructured::CellConstIterator neighIterator = m_surface->getCellConstIterator(neighId);
393
394 normal += computeSegmentNormal(neighIterator);
395 normal /= norm2(normal);
396 }
397
398 return normal;
399}

412std::array<double,3> SegmentationKernel::computeSegmentVertexNormal( const SurfUnstructured::CellConstIterator &segmentIterator, int vertex, bool limited ) const {
413
414 // Segment information
415 long segmentId = segmentIterator.getId();
416 long segmentRawId = segmentIterator.getRawIndex();
417 const Cell &segment = *segmentIterator;
418
419 // Update the cache
420 //
421 // Cache is updated for all the vertices of the segment.
422 long vertexId = segment.getVertexId(vertex);
423 std::size_t vertexRawId = m_surface->getVertices().getRawIndex(vertexId);
424 bool hasUnlimitedNormal = m_unlimitedVertexNormalsValid.rawAt(vertexRawId);
425
426 bool hasLimitedNormal = m_limitedSegmentVertexNormalValid[m_segmentVertexOffset.rawAt(segmentRawId) + vertex];
427
428 if (!hasUnlimitedNormal || !hasLimitedNormal) {
429 static std::vector<long> vertexNeighbours;
430 vertexNeighbours.clear();
431 m_surface->findCellVertexNeighs(segmentId, vertex, &vertexNeighbours);
432
433 std::array<double, 3> limitedVertexNormal;
434 std::array<double, 3> unlimitedVertexNormal;
435 if (hasUnlimitedNormal) {
436 limitedVertexNormal = m_surface->evalLimitedVertexNormal(segmentId, vertex, vertexNeighbours.size(), vertexNeighbours.data(), m_featureAngle) ;
437 unlimitedVertexNormal = m_unlimitedVertexNormalsStorage.rawAt(vertexRawId);
438 } else {
439 m_surface->evalVertexNormals(segmentId, vertex, vertexNeighbours.size(), vertexNeighbours.data(), m_featureAngle, &unlimitedVertexNormal, &limitedVertexNormal) ;
440 }
441
442 // Store vertex limited normal
443 //
444 // Both limited and unlimited normal are evaluated, however limited
445 // normal is only stored if its misalignment with respect to the
446 // unlimited normal is greater than a defined tolerance.
447 if( !hasLimitedNormal ){
448 double misalignment = norm2(unlimitedVertexNormal - limitedVertexNormal) ;
449 if( misalignment >= m_surface->getTol() ){
450 std::pair<long, int> segmentVertexKey = std::make_pair(segmentId, vertex);
451 m_limitedSegmentVertexNormalStorage.insert({segmentVertexKey, std::move(limitedVertexNormal)}) ;
452 }
453 m_limitedSegmentVertexNormalValid[m_segmentVertexOffset.rawAt(segmentRawId) + vertex] = true;
454 }
455
456 // Store vertex unlimited normal
457 if ( !hasUnlimitedNormal ) {
458 m_unlimitedVertexNormalsStorage.rawAt(vertexRawId) = std::move(unlimitedVertexNormal);
459 m_unlimitedVertexNormalsValid.rawAt(vertexRawId) = true ;
460 }
461 }
462
463 // Get the normal from the cache
464 if (limited) {
465 std::pair<long, int> segmentVertexKey = std::make_pair(segmentId, vertex);
466 auto limitedSegmentVertexNormal = m_limitedSegmentVertexNormalStorage.find(segmentVertexKey);
467 if (limitedSegmentVertexNormal != m_limitedSegmentVertexNormalStorage.end()) {
468 return limitedSegmentVertexNormal->second;
469 }
470 }
471
472 return m_unlimitedVertexNormalsStorage.rawAt(vertexRawId);
473}
