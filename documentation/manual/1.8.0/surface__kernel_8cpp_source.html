249double SurfaceKernel::evalCellArea(long id) const
250{
251 // ====================================================================== //
252 // VARIABLES DECLARATION //
253 // ====================================================================== //
254
255 // Local variables
256 const Cell *cell_ = &m_cells[id];
257 ConstProxyVector<long> cellVertexIds = cell_->getVertexIds();
258
259 // Counters
260 // none
261
262 // ====================================================================== //
263 // EVALUATE FACET AREA //
264 // ====================================================================== //
265 if ((cell_->getType() == ElementType::UNDEFINED)
266 || (cell_->getType() == ElementType::VERTEX)) return 0.0;
267 if (cell_->getType() == ElementType::LINE) {
268 return ( norm2(m_vertices[cellVertexIds[0]].getCoords()
269 - m_vertices[cellVertexIds[1]].getCoords()) );
270 }
271 std::array<double, 3> d1, d2;
272 if (cell_->getType() == ElementType::TRIANGLE) {
273 d1 = m_vertices[cellVertexIds[1]].getCoords() - m_vertices[cellVertexIds[0]].getCoords();
274 d2 = m_vertices[cellVertexIds[2]].getCoords() - m_vertices[cellVertexIds[0]].getCoords();
275 return (0.5*norm2(crossProduct(d1, d2)));
276 }
277 else {
278 int nvert = cellVertexIds.size();
279 double coeff = 0.25;
280 double area = 0.0;
281 for (int i = 0; i < nvert; ++i) {
282 int prevVertex = getOrderedLocalVertexIds(*cell_, (nvert + i - 1) % nvert);
283 int vertex = getOrderedLocalVertexIds(*cell_, i);
284 int nextVertex = getOrderedLocalVertexIds(*cell_, (i + 1) % nvert);
285
286 const std::array<double, 3> &prevVertexCoords = m_vertices[cellVertexIds[prevVertex]].getCoords();
287 const std::array<double, 3> &vertexCoords = m_vertices[cellVertexIds[vertex]].getCoords();
288 const std::array<double, 3> &nextVertexCoords = m_vertices[cellVertexIds[nextVertex]].getCoords();
289
290 d1 = nextVertexCoords - vertexCoords;
291 d2 = prevVertexCoords - vertexCoords;
292 area += coeff*norm2(crossProduct(d1, d2));
293 } //next i
294 return(area);
295 }

308double SurfaceKernel::evalEdgeLength(long cellId, int edgeId) const
309{
310 // ====================================================================== //
311 // VARIABLES DECLARATION //
312 // ====================================================================== //
313
314 // Local variables
315 const Cell &cell = m_cells[cellId];
316 double edge_length;
317
318 // Counters
319 // none
320
321 // ====================================================================== //
322 // COMPUTE MIN EDGE SIZE //
323 // ====================================================================== //
324 switch (cell.getType()) {
325
326 case ElementType::LINE:
327 case ElementType::VERTEX:
328 case ElementType::UNDEFINED:
329 {
330 edge_length = 0.;
331 break;
332 }
333
334 default:
335 {
336 ConstProxyVector<long> faceVertexIds = cell.getFaceVertexIds(edgeId);
337 const Vertex &vertex_0 = m_vertices[faceVertexIds[0]];
338 const Vertex &vertex_1 = m_vertices[faceVertexIds[1]];
339 edge_length = norm2(vertex_0.getCoords() - vertex_1.getCoords());
340 break;
341 }
342
343 }
344
345 return edge_length;

445double SurfaceKernel::evalAngleAtVertex(long id, int vertex) const
446{
447 const Cell &cell = m_cells.at(id);
448 ElementType cellType = cell.getType();
449 if (cellType == ElementType::UNDEFINED) {
450 return 0.;
451 } else if (cellType == ElementType::VERTEX) {
452 return 0.;
453 } else if (cellType == ElementType::LINE) {
454 return BITPIT_PI;
455 }
456
457 ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
458 int nCellVertices = cellVertexIds.size();
459
460 int prevVertex = getOrderedLocalVertexIds(cell, (vertex - 1 + nCellVertices) % nCellVertices);
461 int nextVertex = getOrderedLocalVertexIds(cell, (vertex + 1) % nCellVertices);
462
463 const std::array<double, 3> &prevVertexCoords = m_vertices[cellVertexIds[prevVertex]].getCoords();
464 const std::array<double, 3> &vertexCoords = m_vertices[cellVertexIds[vertex]].getCoords();
465 const std::array<double, 3> &nextVertexCoords = m_vertices[cellVertexIds[nextVertex]].getCoords();
466
467 std::array<double, 3> d1 = prevVertexCoords - vertexCoords;
468 d1 = d1 / norm2(d1);
469
470 std::array<double, 3> d2 = nextVertexCoords - vertexCoords;
471 d2 = d2 / norm2(d2);
472
473 double angle = std::acos(std::min(1., std::max(-1., dotProduct(d1, d2))));
474
475 return angle;
476}

570double SurfaceKernel::evalAspectRatio(long id, int &edge_id) const
571{
572 // ====================================================================== //
573 // VARIABLES DECLARATION //
574 // ====================================================================== //
575
576 // Local variables
577 const Cell *cell_ = &m_cells[id];
578
579 // Counters
580 // none
581
582 // ====================================================================== //
583 // EVALUATE ASPECT RATIO //
584 // ====================================================================== //
585 if ((cell_->getType() == ElementType::UNDEFINED)
586 || (cell_->getType() == ElementType::VERTEX)
587 || (cell_->getType() == ElementType::LINE)) return 0.0;
588
589 double l_edge;
590 double m_edge = std::numeric_limits<double>::max();
591 double M_edge = std::numeric_limits<double>::min();
592 int nfaces = cell_->getFaceCount();
593 for (int i = 0; i < nfaces; ++i) {
594 l_edge = evalEdgeLength(id, i);
595 if (l_edge < m_edge) {
596 m_edge = l_edge;
597 edge_id = i;
598 }
599 M_edge = std::max(M_edge, l_edge);
600 } //next i
601
602 return (M_edge/m_edge);

618std::array<double, 3> SurfaceKernel::evalFacetNormal(long id, const std::array<double, 3> &orientation) const
619{
620 // ====================================================================== //
621 // VARIABLES DECLARATION //
622 // ====================================================================== //
623
624 // Local variables
625 std::array<double, 3> normal = {{0.0, 0.0, 0.0}};
626 const Cell *cell_ = &m_cells[id];
627 ConstProxyVector<long> cellVertexIds = cell_->getVertexIds();
628
629 // Counters
630 // none
631
632 // ====================================================================== //
633 // COMPUTE NORMAL //
634 // ====================================================================== //
635 if ((cell_->getType() == ElementType::UNDEFINED)
636 || (cell_->getType() == ElementType::VERTEX)) return normal;
637
638 if (cell_->getType() == ElementType::LINE) {
639 normal = m_vertices[cellVertexIds[1]].getCoords() - m_vertices[cellVertexIds[0]].getCoords();
640 normal = crossProduct(normal, orientation);
641 }
642
643 if (cell_->getType() == ElementType::TRIANGLE) {
644 std::array<double, 3> d1, d2;
645 d1 = m_vertices[cellVertexIds[1]].getCoords() - m_vertices[cellVertexIds[0]].getCoords();
646 d2 = m_vertices[cellVertexIds[2]].getCoords() - m_vertices[cellVertexIds[0]].getCoords();
647 normal = crossProduct(d1, d2);
648 }
649 else {
650 std::array<double, 3> d1, d2;
651 int i, nvert = cellVertexIds.size();
652 double coeff = 1.0/double(nvert);
653 for (i = 0; i < nvert; ++i) {
654 int prevVertex = getOrderedLocalVertexIds(*cell_, (nvert + i - 1) % nvert);
655 int vertex = getOrderedLocalVertexIds(*cell_, i);
656 int nextVertex = getOrderedLocalVertexIds(*cell_, (i + 1) % nvert);
657
658 const std::array<double, 3> &prevVertexCoords = m_vertices[cellVertexIds[prevVertex]].getCoords();
659 const std::array<double, 3> &vertexCoords = m_vertices[cellVertexIds[vertex]].getCoords();
660 const std::array<double, 3> &nextVertexCoords = m_vertices[cellVertexIds[nextVertex]].getCoords();
661
662 d1 = nextVertexCoords - vertexCoords;
663 d2 = prevVertexCoords - vertexCoords;
664 normal += coeff*crossProduct(d1, d2);
665 } //next i
666 }
667 normal = normal/norm2(normal);
668
669 return(normal);
670
671}

683{
684 // ====================================================================== //
685 // VARIABLES DECLARATION //

700 } //next i
701 normal = normal/norm2(normal);
702 }
703 return(normal);

822void SurfaceKernel::evalVertexNormals(long id, int vertex, std::size_t nVertexNeighs, const long *vertexNeighs, double limit,
823 std::array<double, 3> *unlimitedNormal, std::array<double, 3> *limitedNormal) const
824{
825 // Early return if no calculation is needed
826 if (!unlimitedNormal && !limitedNormal) {
827 return;
828 }
829
830 // Get vertex id
831 const Cell &cell = getCell(id);
832 long vertexId = cell.getVertexId(vertex);
833
834 // Get cell information
835 double cellVertexAngle = evalAngleAtVertex(id, vertex);
836 std::array<double, 3> cellNormal = evalFacetNormal(id);
837
838 // Initialize unlimited normal
839 if (unlimitedNormal) {
840 *unlimitedNormal = cellVertexAngle * cellNormal;
841 }
842
843 // Initialize limited normal
844 if (limitedNormal) {
845 *limitedNormal = cellVertexAngle * cellNormal;
846 }
847
848 // Add contribution of neighbouring cells
849 for (std::size_t i = 0; i < nVertexNeighs; ++i) {
850 // Get neighbour information
851 long neighId = vertexNeighs[i];
852 const Cell &neigh = getCell(neighId);
853 std::array<double, 3> neighNormal = evalFacetNormal(neighId);
854 double neighVertexAngle = evalAngleAtVertex(neighId, neigh.findVertex(vertexId));
855
856 // Add contribution to unlimited normal
857 if (unlimitedNormal) {
858 *unlimitedNormal += neighVertexAngle * neighNormal;
859 }
860
861 // Add contribution to limited normal
862 //
863 // Only the negihbours whose normal has a misalignment less then the
864 // specified limit are considered.
865 if (limitedNormal) {
866 // The argument of the acos function has to be in the range [-1, 1].
867 // Rounding errors may lead to a dot product slightly outside this
868 // range. Since the arguments of the dot product are unit vectors,
869 // we can safetly clamp the dot product result to be between -1 and
870 // 1.
871 double misalignment = std::acos(std::min(1.0, std::max(-1.0, dotProduct(neighNormal, cellNormal))));
872 if (misalignment <= std::abs(limit)) {
873 *limitedNormal += neighVertexAngle * neighNormal;
874 }
875 }
876 }
877
878 // Normalize the unlimited normal
879 if (unlimitedNormal) {
880 *unlimitedNormal /= norm2(*unlimitedNormal);
881 }
882
883 // Normalize the unlimited normal
884 if (limitedNormal) {
885 *limitedNormal /= norm2(*limitedNormal);
886 }
887}

900bool SurfaceKernel::adjustCellOrientation()
901{
902 // Early return if the patch is empty.
903 if (empty()) {
904 return false;
905 }
906
907 // Get the first facet
908 long firstFacetId = Cell::NULL_ID;
909 if (getCellCount() > 0) {
910 firstFacetId = getCells().begin()->getId();
911 }
912
913#if BITPIT_ENABLE_MPI==1
914 if (isPartitioned()) {
915 int patchRank = getRank();
916
917 int firstFacetRank = std::numeric_limits<int>::max();
918 if (firstFacetId >= 0) {
919 firstFacetRank = patchRank;
920 }
921
922 MPI_Allreduce(MPI_IN_PLACE, &firstFacetRank, 1, MPI_INT, MPI_MIN, getCommunicator());
923 if (patchRank != firstFacetRank) {
924 firstFacetId = Cell::NULL_ID;
925 }
926 }
927#endif
928
929 // Adjust cell orientation according to the orientation of the first facet
930 return adjustCellOrientation(firstFacetId);
931}

950bool SurfaceKernel::adjustCellOrientation(long seed, bool invert)
951{
952#if BITPIT_ENABLE_MPI==1
953 // Check if the communications are needed
954 bool ghostExchangeNeeded = isPartitioned();
955
956 // Initialize ghost communications
957 std::unordered_set<long> flipped;
958 std::unique_ptr<DataCommunicator> ghostComm;
959 if (ghostExchangeNeeded) {
960 ghostComm = std::unique_ptr<DataCommunicator>(new DataCommunicator(getCommunicator()));
961
962 for (auto &entry : getGhostCellExchangeSources()) {
963 const int rank = entry.first;
964 const std::vector<long> &sources = entry.second;
965 ghostComm->setSend(rank, sources.size() * sizeof(bool));
966 }
967
968 for (auto &entry : getGhostCellExchangeTargets()) {
969 const int rank = entry.first;
970 const std::vector<long> &targets = entry.second;
971 ghostComm->setRecv(rank, targets.size() * sizeof(bool));
972 }
973 }
974#endif
975
976 // Initialize the seed
977 std::set<long> toVisit;
978 if (seed != Element::NULL_ID) {
979 assert(getCells().exists(seed));
980 toVisit.insert(seed);
981 if (invert) {
982 flipCellOrientation(seed);
983#if BITPIT_ENABLE_MPI==1
984 if (ghostExchangeNeeded) {
985 flipped.insert(seed);
986 }
987#endif
988 }
989 }
990
991 // Orient the surface
992 std::unordered_set<long> visited;
993
994 bool orientable = true;
995 bool completed = false;
996 while (!completed) {
997 while (orientable && !toVisit.empty()) {
998 auto toVisitBegin = toVisit.begin();
999 long cellId = *toVisitBegin;
1000 toVisit.erase(toVisitBegin);
1001
1002 visited.insert(cellId);
1003
1004 const auto &cell = getCell(cellId);
1005 int nCellFaces = cell.getFaceCount();
1006 for (int face = 0; face < nCellFaces; ++face) {
1007 int nFaceAdjacencies = cell.getAdjacencyCount(face);
1008 const long *faceAdjacencies = cell.getAdjacencies(face);
1009 for (int i = 0; i < nFaceAdjacencies; ++i) {
1010 long neighId = faceAdjacencies[i];
1011 const Cell &neigh = getCell(neighId);
1012#if BITPIT_ENABLE_MPI==1
1013 if (ghostExchangeNeeded) {
1014 if (!neigh.isInterior()) {
1015 continue;
1016 }
1017 }
1018#endif
1019
1020 int neighFace = findAdjoinNeighFace(cell, face, neigh);
1021
1022 bool isNeighOriented = haveSameOrientation(cell, face, neigh, neighFace);
1023 if (visited.count(neighId) == 0) {
1024 if (!isNeighOriented) {
1025 flipCellOrientation(neighId);
1026#if BITPIT_ENABLE_MPI==1
1027 if (ghostExchangeNeeded) {
1028 flipped.insert(neighId);
1029 }
1030#endif
1031 }
1032
1033 toVisit.insert(neighId);
1034 } else if (!isNeighOriented) {
1035 orientable = false;
1036 break;
1037 }
1038
1039 if (!orientable) {
1040 break;
1041 }
1042 }
1043 }
1044 }
1045
1046#if BITPIT_ENABLE_MPI==1
1047 // Comunicate the orientable flag among the partitions
1048 if (ghostExchangeNeeded) {
1049 MPI_Allreduce(MPI_IN_PLACE, &orientable, 1, MPI_C_BOOL, MPI_LAND, getCommunicator());
1050 }
1051#endif
1052
1053 // If the surface is not orientable we can exit
1054 if (!orientable) {
1055 break;
1056 }
1057
1058#if BITPIT_ENABLE_MPI==1
1059 if (ghostExchangeNeeded) {
1060 // Add seeds from other partitions
1061 ghostComm->startAllRecvs();
1062
1063 for(auto &entry : getGhostCellExchangeSources()) {
1064 int rank = entry.first;
1065 SendBuffer &buffer = ghostComm->getSendBuffer(rank);

1098#endif

1133bool SurfaceKernel::haveSameOrientation(const Cell &cell_A, int face_A, const Cell &cell_B, int face_B) const
1134{
1135 //
1136 // Cell information
1137 //
1138 long cellId_A = cell_A.getId();
1139 ConstProxyVector<long> cellVertexIds_A = cell_A.getVertexIds();
1140 std::size_t nCellVertices_A = cellVertexIds_A.size();
1141
1142 long cellId_B = cell_B.getId();
1143 ConstProxyVector<long> cellVertexIds_B = cell_B.getVertexIds();
1144 std::size_t nCellVertices_B = cellVertexIds_B.size();
1145
1146 assert(cell_A.getDimension() == cell_B.getDimension());
1147 int cellDimension = cell_A.getDimension();
1148 assert(cellDimension < 3);
1149
1150 //
1151 // Check relative orientation using vertices
1152 //
1153
1154 // If the facets share at least two consecutive vertices, it is possible
1155 // to detect relative orientation checking the order in which these two
1156 // vertices appear while traversing the list of vertices of the facets.
1157 // If the vertices appear in the same order, the facets have opposite
1158 // orientation, if the vertices appear in reversed order, the facets have
1159 // the same orientation.
1160 for (std::size_t i = 0; i < nCellVertices_A; ++i) {
1161 long vertexId_A = cellVertexIds_A[getOrderedLocalVertexIds(cell_A, i)];
1162 for (std::size_t j = 0; j < nCellVertices_B; ++j) {
1163 long vertexId_B = cellVertexIds_B[getOrderedLocalVertexIds(cell_B, j)];
1164 if (vertexId_A == vertexId_B) {
1165 if (cellDimension == 2) {
1166 long previousVertexId_A = cellVertexIds_A[getOrderedLocalVertexIds(cell_A, (i - 1 + nCellVertices_A) % nCellVertices_A)];
1167 long previousVertexId_B = cellVertexIds_B[getOrderedLocalVertexIds(cell_B, (j - 1 + nCellVertices_B) % nCellVertices_B)];
1168
1169 long nextVertexId_A = cellVertexIds_A[getOrderedLocalVertexIds(cell_A, (i + 1) % nCellVertices_A)];
1170 long nextVertexId_B = cellVertexIds_B[getOrderedLocalVertexIds(cell_B, (j + 1) % nCellVertices_B)];
1171
1172 if (nextVertexId_A == nextVertexId_B || previousVertexId_A == previousVertexId_B) {
1173 return false;
1174 } else if (nextVertexId_A == previousVertexId_B || previousVertexId_A == nextVertexId_B) {
1175 return true;
1176 }
1177 } else if (cellDimension == 1) {
1178 return (i != j);
1179 } else {
1180 throw std::runtime_error("Unable to identify if the factes has the same orientation.");
1181 }
1182 }
1183 }
1184 }
1185
1186 //
1187 // Check relative orientation using facet normals
1188 //
1189
1190 // The faces doesn't have enough common vertices to check the orientation
1191 // using topological checks, we need a geometric check. Starting from the
1192 // first vertex of the face, we identify three non-collinear vertices and
1193 // we detect their winding. The two facets have the same orientation if
1194 // they have the same winding.
1195 //
1196 // To detect the winding of the vertices, we evaluate the normal of the
1197 // plane defined by the vertices and then we calculate the projection of
1198 // that vector along a direction (called winding direction) which is
1199 // chosen equal for both the cells. The sign of the projection defines
1200 // the winding of the vertices. We don't care if the winding of a single
1201 // cell is clockwise or counter-clockwise, all we care about is if the
1202 // two cells have the same winding.
1203
1204 // Pseudo normal directions
1205 //
1206 // For each face, we identify three non-collinear vertices and we evaluate
1207 // the normal direction of the plane defined by those three vertices.
1208 std::size_t nMaxCellVertices = std::max(nCellVertices_A, nCellVertices_B);
1209 BITPIT_CREATE_WORKSPACE(vertexCoordinates, std::array<double BITPIT_COMMA 3>, nMaxCellVertices, ReferenceElementInfo::MAX_ELEM_VERTICES);
1210
1211 ConstProxyVector<int> faceLocalVertexIds_A = cell_A.getFaceLocalVertexIds(face_A);
1212 ConstProxyVector<int> faceLocalVertexIds_B = cell_B.getFaceLocalVertexIds(face_B);
1213
1214 std::array<double, 3> pseudoNormal_A;
1215 std::array<double, 3> pseudoNormal_B;
1216 for (int i = 0; i < 2; ++i) {
1217 std::array<double, 3> *pseudoNormal;
1218 long firstFaceVertexLocalId;
1219 std::size_t nCellVertices;
1220 if (i == 0) {
1221 pseudoNormal = &pseudoNormal_A;
1222 firstFaceVertexLocalId = faceLocalVertexIds_A[0];
1223 nCellVertices = nCellVertices_A;
1224 getCellVertexCoordinates(cellId_A, vertexCoordinates);
1225 } else {
1226 pseudoNormal = &pseudoNormal_B;
1227 firstFaceVertexLocalId = faceLocalVertexIds_B[0];
1228 nCellVertices = nCellVertices_B;
1229 getCellVertexCoordinates(cellId_B, vertexCoordinates);
1230 }
1231 assert(nCellVertices >= 3);
1232
1233 const std::array<double, 3> &V_0 = vertexCoordinates[(firstFaceVertexLocalId + 0) % nCellVertices];
1234 const std::array<double, 3> &V_1 = vertexCoordinates[(firstFaceVertexLocalId + 1) % nCellVertices];
1235 for (std::size_t k = 2; k < nCellVertices; ++k) {
1236 const std::array<double, 3> &V_2 = vertexCoordinates[(firstFaceVertexLocalId + k) % nCellVertices];
1237
1238 *pseudoNormal = crossProduct(V_0 - V_1, V_2 - V_1);
1239 if (!utils::DoubleFloatingEqual()(norm2(*pseudoNormal), 0.)) {
1240 break;
1241 } else if (k == (nCellVertices - 1)) {
1242 throw std::runtime_error("Unable to identify the non-collinear vertices.");
1243 }
1244 }
1245 }
1246
1247 // Direction for evaluating the winding
1248 //
1249 // This direction has to be the same for both the cells and should not be
1250 // perpendicular to the pseudo normals.
1251 std::array<double, 3> windingDirection = {{0., 0., 1.}};
1252 for (int d = 2; d <= 0; --d) {
1253 windingDirection = {{0., 0., 0.}};
1254 windingDirection[d] = 1.;
1255
1256 double pseudNormalProjection_A = dotProduct(pseudoNormal_A, windingDirection);
1257 double pseudNormalProjection_B = dotProduct(pseudoNormal_B, windingDirection);
1258 if (!utils::DoubleFloatingEqual()(pseudNormalProjection_A, 0.) && !utils::DoubleFloatingEqual()(pseudNormalProjection_B, 0.)) {
1259 break;
1260 } else if (d == 0) {
1261 throw std::runtime_error("Unable to identify the relative orientation of the cells.");
1262 }
1263 }
1264
1265 // Identify the winding of the cells
1266 int cellWinding_A = (dotProduct(pseudoNormal_A, windingDirection) > 0.);
1267 int cellWinding_B = (dotProduct(pseudoNormal_B, windingDirection) > 0.);
1268
1269 return (cellWinding_A == cellWinding_B);
1270}

1277void SurfaceKernel::flipCellOrientation(long id)
1278{
1279 // Cell information
1280 Cell &cell = getCell(id);
1281
1282 //
1283 // Flip connectivity
1284 //
1285 // In order to flip cell orientation, we need to reverse the order of
1286 // the vertices:
1287 //
1288 // Vertices 0 1 2 3 4
1289 //
1290 // Flipped vertices 4 3 2 1 0
1291 //
1292 // This means we have to invert the connectivity of the cell.
1293 long *connectivity = cell.getConnect();
1294
1295 int connectBegin = 0;
1296 if (cell.getType() == ElementType::POLYGON) {
1297 ++connectBegin;
1298 }
1299
1300 int nCellVertices = cell.getVertexCount();
1301 for (int i = 0; i < (int) std::floor(nCellVertices / 2); ++i) {
1302 std::swap(connectivity[connectBegin + i], connectivity[connectBegin + nCellVertices - i - 1]);
1303 }
1304
1305 //
1306 // Flip adjacencies and interfaces
1307 //
1308 // Adjacencies and interfaces are associated with the faces of the cells,
1309 // since we have flipped the connectivity we need to flip also adjacencies
1310 // and interfaces.
1311 //
1312 // For two-dimensional elements faces are defined between two vertices:
1313 //
1314 // Vertices 0 1 2 3 4
1315 // Faces 0 1 2 3 4
1316 //
1317 // Flipped vertices 4 3 2 1 0
1318 // Flipped faces 3 2 1 0 4
1319 //
1320 // We have reversed the order of the vertices, this means that we have
1321 // to invert the order of all the faces but the lust one.
1322 //
1323 // For one-dimensional elements faces are defined on the vertices itself:
1324 //
1325 // Vertices 0 1
1326 // Faces 0 1
1327 //
1328 // Flipped vertices 1 0
1329 // Flipped faces 1 0
1330 //
1331 // We have reversed the order of the vertices, this means that we have
1332 // to invert the order of all the faces.
1333 int nCellFaces = cell.getFaceCount();
1334 int nCellAdjacencies = cell.getAdjacencyCount();
1335 int nCellInterfaces = cell.getInterfaceCount();
1336
1337 int flipOffset;
1338 if (cell.getDimension() == 1) {
1339 flipOffset = 0;
1340 } else {
1341 flipOffset = 1;
1342 }
1343
1344 FlatVector2D<long> flippedCellAdjacencies;
1345 FlatVector2D<long> flippedCellInterfaces;
1346 flippedCellAdjacencies.reserve(nCellFaces, nCellAdjacencies);
1347 flippedCellInterfaces.reserve(nCellFaces, nCellInterfaces);
1348
1349 for (int i = 0; i < nCellFaces - flipOffset; ++i) {
1350 int nFaceAdjacencies = cell.getAdjacencyCount(nCellFaces - i - 1 - flipOffset);
1351 const long *faceAdjacencies = cell.getAdjacencies(nCellFaces - i - 1 - flipOffset);
1352 flippedCellAdjacencies.pushBack(nFaceAdjacencies, faceAdjacencies);
1353
1354 int nFaceInterfaces = cell.getInterfaceCount(nCellFaces - i - 1 - flipOffset);
1355 const long *faceInterfaces = cell.getInterfaces(nCellFaces - i - 1 - flipOffset);
1356 flippedCellInterfaces.pushBack(nFaceInterfaces, faceInterfaces);
1357 }
1358
1359 if (flipOffset == 1) {
1360 int nLastFaceAdjacencies = cell.getAdjacencyCount(nCellFaces - 1);
1361 const long *lastFaceAdjacencies = cell.getAdjacencies(nCellFaces - 1);
1362 flippedCellAdjacencies.pushBack(nLastFaceAdjacencies, lastFaceAdjacencies);
1363
1364 int nLastFaceInterfaces = cell.getInterfaceCount(nCellFaces - 1);
1365 const long *lastFaceInterfaces = cell.getInterfaces(nCellFaces - 1);
1366 flippedCellInterfaces.pushBack(nLastFaceInterfaces, lastFaceInterfaces);
1367 }
1368
1369 // Set flipped adjacencies
1370 if (nCellAdjacencies > 0) {
1371 cell.setAdjacencies(std::move(flippedCellAdjacencies));
1372 }
1373
1374 // Set flipped interfaces
1375 //
1376 // After setting the flipped interfaces, we also need to update the face
1377 // associated with the interfaces.
1378 if (nCellInterfaces > 0) {
1379 cell.setInterfaces(std::move(flippedCellInterfaces));
1380
1381 const long *cellInterfaces = cell.getInterfaces();
1382 int nCellInterfaces = cell.getInterfaceCount();
1383 for (int i = 0; i < nCellInterfaces; ++i) {
1384 long interfaceId = cellInterfaces[i];
1385 Interface &interface = getInterface(interfaceId);
1386
1387 long owner = interface.getOwner();
1388 if (owner == id) {
1389 int ownerFace = interface.getOwnerFace();
1390 if (ownerFace != nCellFaces - 1) {
1391 ownerFace = nCellFaces - 2 - ownerFace;
1392 interface.setOwner(id, ownerFace);
1393 }
1394 } else {
1395 int neighFace = interface.getNeighFace();
1396 if (neighFace != nCellFaces - 1) {
1397 neighFace = nCellFaces - 2 - neighFace;
1398 interface.setNeigh(id,neighFace);
1399 }
1400 }
1401 }
1402 }

1503 // ====================================================================== //
