41void VolOctree::initializeTreePartitioning()
42{
43 // Move all the octants to the first processor
44 //
45 // Assigning a null weight to every octant but the last octant and then
46 // doing a load balance will have the effect of moving all the octants
47 // to the first process.
48 std::size_t nOctants = m_tree->getNumOctants();
49 std::vector<double> octantWeights(nOctants, 0.);
50 if (nOctants > 0) {
51 octantWeights[nOctants - 1] = 1.;
52 }
53 m_tree->loadBalance(m_partitioningOctantWeights.get());
54}

137std::vector<adaption::Info> VolOctree::_partitioningPrepare(const std::unordered_map<long, double> &cellWeights, double defaultWeight, bool trackPartitioning)
138{
139 // Eval partitioning weights
140 computePartitioningOctantWeights(cellWeights, defaultWeight);
141
142 // Generate partitioning information
143 std::vector<adaption::Info> partitioningData;
144 if (trackPartitioning) {
145 int currentRank = getRank();
146 PabloUniform::LoadBalanceRanges loadBalanceRanges = m_tree->evalLoadBalanceRanges(m_partitioningOctantWeights.get());
147 for (const auto &entry : loadBalanceRanges.sendRanges) {
148 int receiver = entry.first;
149 if (receiver == currentRank) {
150 continue;
151 }
152
153 adaption::Type adaptionType;
154 if (loadBalanceRanges.sendAction == PabloUniform::LoadBalanceRanges::ACTION_DELETE) {
155 adaptionType = adaption::TYPE_DELETION;
156 } else {
157 adaptionType = adaption::TYPE_PARTITION_SEND;
158 }
159
160 partitioningData.emplace_back();
161 adaption::Info &partitioningInfo = partitioningData.back();
162 partitioningInfo.entity = adaption::ENTITY_CELL;
163 partitioningInfo.type = adaptionType;
164 partitioningInfo.rank = receiver;
165
166 const std::array<uint32_t, 2> &sendRange = entry.second;
167 uint32_t beginTreeId = sendRange[0];
168 uint32_t endTreeId = sendRange[1];
169 partitioningInfo.previous.reserve(endTreeId - beginTreeId);
170 for (uint32_t treeId = beginTreeId; treeId < endTreeId; ++treeId) {
171 OctantInfo octantInfo(treeId, true);
172 partitioningInfo.previous.emplace_back();
173 long &cellId = partitioningInfo.previous.back();
174 cellId = getOctantId(octantInfo);
175 }
176 }
177 }
178
179 return partitioningData;
180}

193std::vector<adaption::Info> VolOctree::_partitioningAlter(bool trackPartitioning)
194{
195 std::vector<adaption::Info> partitioningData;
196
197 // Early return if the dimension of the tree is null
198 if (m_tree->getDim() == 0) {
199 return partitioningData;
200 }
201
202 // Updating the tree
203 m_tree->loadBalance(m_partitioningOctantWeights.get());
204
205 // Sync the patch
206 partitioningData = sync(trackPartitioning);
207
208 // The bounding box is frozen, it is not updated automatically
209 setBoundingBox();
210
211 return partitioningData;
212}

230std::vector<long> VolOctree::_findGhostCellExchangeSources(int rank)
231{
232 const std::vector<uint32_t> &ghostOctantSources = m_tree->getBordersPerProc().at(rank);
233 std::size_t nGhostSources = ghostOctantSources.size();
234
235 std::vector<long> ghostCellSources(nGhostSources);
236 for (std::size_t k = 0; k < nGhostSources; ++k) {
237 OctantInfo octantInfo(ghostOctantSources[k], true);
238 ghostCellSources[k] = getOctantId(octantInfo);
239 }
240
241 return ghostCellSources;
242}

253void VolOctree::computePartitioningOctantWeights(const std::unordered_map<long, double> &cellWeights, double defaultWeight)
254{
255 bool useWeights = !cellWeights.empty();
256 MPI_Allreduce(MPI_IN_PLACE, &useWeights, 1, MPI_C_BOOL, MPI_LOR, getCommunicator());
257 if (!useWeights) {
258 clearPartitioningOctantWeights();
259 return;
260 }
261
262 CellConstIterator beginItr = internalCellConstBegin();
263 CellConstIterator endItr = internalCellConstEnd();
264
265 std::size_t nOctants = m_tree->getNumOctants();
266 m_partitioningOctantWeights = std::unique_ptr<std::vector<double>>(new std::vector<double>(nOctants, defaultWeight));
267 for (CellConstIterator cellItr = beginItr; cellItr != endItr; ++cellItr) {
268 long cellId = cellItr.getId();
269 auto weightItr = cellWeights.find(cellId);
270 if (weightItr != cellWeights.end()) {
271 OctantInfo octantInfo = getCellOctant(cellId);
272 uint32_t treeId = octantInfo.id;
273 (*m_partitioningOctantWeights)[treeId] = weightItr->second;
274 }
275 }
276}
