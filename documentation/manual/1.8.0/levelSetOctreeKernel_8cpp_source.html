129void LevelSetOctreeKernel::updateCellCirclesCache( ) {
130
131 int dimension = m_octree->getDimension();
132 int maxLevel = m_octree->getTree().getMaxLevel();
133
134 m_levelToCellIncircle.resize(maxLevel + 1);
135 m_levelToCellCircumcircle.resize(maxLevel + 1);
136 for (int level = 0; level <= maxLevel; ++level) {
137 double levelSize = m_octree->getTree().levelToSize(level);
138
139 m_levelToCellIncircle[level] = 0.5 * levelSize;
140 m_levelToCellCircumcircle[level] = 0.5 * std::sqrt(dimension) * levelSize;
141 }
142
143}
