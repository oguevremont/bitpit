163std::string FileHandler::getPath(){
164 std::stringstream filename ;
165
166 if (!directory.empty()) filename << directory << "/";
167 filename << name;
168 if(parallel) filename <<".b"<< bitpit::utils::string::zeroPadNumber(4, block) ;
169 if(series) filename <<"." << bitpit::utils::string::zeroPadNumber(4, counter) ;
170 filename <<"."<< appendix ;
171
172 return filename.str() ;
173
174};
