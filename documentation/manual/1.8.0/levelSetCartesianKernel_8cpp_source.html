152void LevelSetCartesianKernel::updateCellCirclesCache( ) {
153
154 int dimension = m_cartesian->getDimension();
155 std::array<double,3> spacing = m_cartesian->getSpacing();
156
157 m_cellIncircle = std::numeric_limits<double>::max();
158 m_cellCircumcircle = 0.;
159 for(int i=0; i<dimension; ++i){
160 m_cellIncircle = std::min(m_cellIncircle, spacing[i]);
161 m_cellCircumcircle += spacing[i] * spacing[i];
162 }
163 m_cellIncircle = 0.5 * m_cellIncircle;
164 m_cellCircumcircle = 0.5 * std::sqrt(m_cellCircumcircle);
165
166}
