35template<typename weight_t>
36bitpit::OBinaryStream& operator<<(bitpit::OBinaryStream &buffer, const bitpit::DiscreteStencil<weight_t> &stencil)
37{
38 buffer << stencil.m_zero;
39
40 std::size_t nItems = stencil.size();
41 buffer << nItems;
42
43 const long *patternData = stencil.patternData();
44 const weight_t *weightData = stencil.weightData();
45 for (std::size_t n = 0; n < nItems; ++n) {
46 buffer << patternData[n];
47 buffer << weightData[n];
48 }
49
50 buffer << stencil.m_constant;
51
52 return buffer;
53}

588template<typename weight_t>
589void DiscreteStencil<weight_t>::optimize(double tolerance)
590{
591 std::size_t nItems = size();
592 for (std::size_t n = 0; n < nItems; ++n) {
593 bool isWeightNeglibile = optimizeWeight(n, tolerance);
594 if (isWeightNeglibile) {
595 m_pattern.erase(m_pattern.begin() + n);
596 m_weights.erase(m_weights.begin() + n);
597 --n;
598 --nItems;
599 }
600 }
601}

623template<typename weight_t>
624void DiscreteStencil<weight_t>::addComplementToZero(long id)
625{
626 const std::size_t nItems = size();
627 if (nItems == 0) {
628 return;
629 }
630
631 weight_t complement = -1. * m_weights[0];
632 for (std::size_t n = 1; n < nItems; ++n) {
633 complement -= m_weights[n];
634 }
635
636 appendItem(id, complement);
637}

736template<typename weight_t>
737void DiscreteStencil<weight_t>::display(std::ostream &out, double factor) const
738{
739 const std::size_t nItems = size();
740
741 weight_t sum = m_zero;
742 for (std::size_t n = 0; n < nItems; ++n) {
743 long id = m_pattern[n];
744 weight_t value = factor * m_weights[n];
745 out << " id: " << id << " weight: " << value << std::endl;
746 sum += value;
747 }
748
749 out << " constant : " << (factor * m_constant) << std::endl;
750 out << " sum : " << sum << std::endl;
751}
