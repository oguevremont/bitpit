255Config::MultiSection Config::getSections(const std::string &key)
256{
257 MultiSection sections;
258 sections.reserve(getSectionCount(key));
259
260 auto range = m_sections->equal_range(key);
261 for (auto itr = range.first; itr != range.second; ++itr) {
262 sections.push_back(itr->second.get());
263 }
264
265 return sections;
266}

319const Config::Section & Config::getSection(const std::string &key) const
320{
321 auto sectionItr = m_sections->find(key);
322 if (sectionItr == m_sections->end()) {
323 throw std::runtime_error("The section named \"" + key + "\" does not esists");
324 }
325
326 return *(sectionItr->second);

337Config::Section & Config::addSection(const std::string &key)
338{
339 if (!m_multiSections && hasSection(key)) {
340 throw std::runtime_error("A section named \"" + key + "\" already esists");
341 }
342
343 std::unique_ptr<Section> section = std::unique_ptr<Section>(new Section(m_multiSections));
344 auto sectionItr = m_sections->insert(std::make_pair(key, std::move(section)));
345
346 return *(sectionItr->second);
347}

401void Config::dump(std::ostream &out, int indentLevel) const
402{
403 const int INDENT_SIZE = 2;
404
405 std::string padding(INDENT_SIZE, ' ');
406 std::string indent(INDENT_SIZE * indentLevel, ' ');
407
408 out << std::endl;
409 out << indent << "Options..." << std::endl;
410 if (getOptionCount() > 0) {
411 for (auto &entry : getOptions()) {
412 out << indent << padding << entry.first << " = " << entry.second << std::endl;
413 }
414 } else {
415 out << indent << padding << "No options." << std::endl;
416 }
417
418 ++indentLevel;
419 for (auto &entry : getSections()) {
420 out << std::endl;
421 out << indent << padding << "::: Section " << entry.first << " :::" << std::endl;
422 entry.second->dump(out, indentLevel);
423 }
424}
