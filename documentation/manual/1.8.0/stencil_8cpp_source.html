75template<>
76bool DiscreteStencil<std::array<double, 3>>::optimizeWeight(std::size_t pos, double tolerance)
77{
78 std::array<double, 3> &weight = m_weights[pos];
79 int nItems = weight.size();
80
81 int nNegligibleItems = 0;
82 for (int k = 0; k < nItems; ++k) {
83 if (std::abs(weight[k] - m_zero[k]) > tolerance) {
84 continue;
85 }
86
87 weight[k] = m_zero[k];
88 ++nNegligibleItems;
89 }
90
91 return (nNegligibleItems == nItems);
92}

141bitpit::StencilVector operator*(const std::array<double, 3> &vector, const bitpit::StencilScalar &stencil)
142{
143 const std::size_t nItems = stencil.size();
144 bitpit::StencilVector stencil_B;
145 stencil_B.resize(nItems);
146 for (std::size_t n = 0; n < nItems; ++n) {
147 stencil_B.setPattern(n, stencil.getPattern(n));
148 stencil_B.setWeight(n, ::operator*(stencil.getWeight(n), vector));
149 }
150 stencil_B.setConstant(::operator*(stencil.getConstant(), vector));
151
152 return stencil_B;
153}

177void dotProduct(const bitpit::StencilVector &stencil, const bitpit::StencilVector::weight_type &vector, bitpit::StencilScalar *stencil_dotProduct)
178{
179 const std::size_t nItems = stencil.size();
180 stencil_dotProduct->resize(nItems);
181
182 const long *patternData = stencil.patternData();
183 const bitpit::StencilVector::weight_type *weightData = stencil.weightData();
184 long *patternData_dotProduct = stencil_dotProduct->patternData();
185 bitpit::StencilScalar::weight_type *weightData_dotProduct = stencil_dotProduct->weightData();
186 for (std::size_t n = 0; n < nItems; ++n) {
187 patternData_dotProduct[n] = patternData[n];
188 weightData_dotProduct[n] = ::dotProduct(weightData[n], vector);
189 }
190
191 stencil_dotProduct->setConstant(::dotProduct(stencil.getConstant(), vector));
192}

216void project(const bitpit::StencilVector &stencil, const std::array<double, 3> &direction, bitpit::StencilVector *stencil_projection)
217{
218 stencil_projection->initialize(stencil);
219 const std::size_t nItems = stencil_projection->size();
220
221 bitpit::StencilVector::weight_type *weightData_projection = stencil_projection->weightData();
222 for (std::size_t n = 0; n < nItems; ++n) {
223 bitpit::StencilVector::weight_type &weight = weightData_projection[n];
224 weight = ::dotProduct(weight, direction) * direction;
225 }
226
227 stencil_projection->setConstant(::dotProduct(stencil_projection->getConstant(), direction) * direction);
228}
