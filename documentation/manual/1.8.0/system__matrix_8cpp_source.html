227void SparseMatrix::_initialize(bool partitioned, long nRows, long nCols, long nNZ)
228{
229 // Serial initialization
230 _initialize(nRows, nCols, nNZ);
231
232 // Parallel initialization
233 m_partitioned = partitioned;
234
235 m_global_nRows = m_nRows;
236 m_global_nCols = m_nCols;
237 if (m_partitioned) {
238 MPI_Allreduce(MPI_IN_PLACE, &m_global_nRows, 1, MPI_LONG, MPI_SUM, m_communicator);
239 MPI_Allreduce(MPI_IN_PLACE, &m_global_nCols, 1, MPI_LONG, MPI_SUM, m_communicator);
240 }
241
242 m_global_rowOffset = 0;
243 m_global_colOffset = 0;
244 if (m_partitioned) {
245 int nProcessors;
246 MPI_Comm_size(m_communicator, &nProcessors);
247
248 std::vector<long> nGlobalRows(nProcessors);
249 MPI_Allgather(&nRows, 1, MPI_LONG, nGlobalRows.data(), 1, MPI_LONG, m_communicator);
250
251 std::vector<long> nGlobalCols(nProcessors);
252 MPI_Allgather(&nCols, 1, MPI_LONG, nGlobalCols.data(), 1, MPI_LONG, m_communicator);
253
254 int rank;
255 MPI_Comm_rank(m_communicator, &rank);
256 for (int i = 0; i < rank; ++i) {
257 m_global_rowOffset += nGlobalRows[i];
258 m_global_colOffset += nGlobalCols[i];
259 }
260 }
261}
262#endif

346void SparseMatrix::assembly()
347{
348 // Early return if the matrix is already assembled.
349 if (isAssembled()) {
350 return;
351 }
352
353 // Assembly can be called only after adding all the rows
354 if (countMissingRows() != 0) {
355 throw std::runtime_error("Assembly can be called only after adding all the rows.");
356 }
357
358#if BITPIT_ENABLE_MPI==1
359 // Updathe global information of the non-zero elements
360 if (m_partitioned) {
361 MPI_Allreduce(&m_maxRowNZ, &m_global_maxRowNZ, 1, MPI_LONG, MPI_MAX, m_communicator);
362 MPI_Allreduce(&m_nNZ, &m_global_nNZ, 1, MPI_LONG, MPI_SUM, m_communicator);
363 }
364#endif
365
366 // The function is now assembled
367 m_assembled = true;
368}

568std::vector<long> SparseMatrix::extractLocalGlobalRows() const
569{
570 const std::size_t *rowExtents = m_pattern.indices();
571
572 std::vector<long> localGlobalRows;
573 localGlobalRows.reserve(m_nRows);
574 for (long i = 0; i < m_nRows; ++i) {
575 std::size_t nRowNZ = rowExtents[i + 1] - rowExtents[i];
576 if (nRowNZ > 0) {
577 localGlobalRows.push_back(m_global_rowOffset + i);
578 }
579 }
580
581 return localGlobalRows;
582}

599std::vector<long> SparseMatrix::extractLocalGlobalCols() const
600{
601 long firstGlobalCol = m_global_colOffset;
602 long lastGlobalCol = m_global_colOffset + m_nCols - 1;
603
604 const long *globalCols = m_pattern.data();
605
606 std::vector<long> localGlobalCols;
607 localGlobalCols.reserve(m_nNZ);
608 for (long k = 0; k < m_nNZ; ++k) {
609 long globalCol = globalCols[k];
610 if (globalCol < firstGlobalCol || globalCol >= lastGlobalCol) {
611 utils::addToOrderedVector<long>(globalCol, localGlobalCols);
612 }
613 }
614
615 return localGlobalCols;
616}

663void SparseMatrix::addRow(long nRowNZ, const long *rowPattern, const double *rowValues)
664{
665 if (countMissingRows() == 0) {
666 throw std::runtime_error("Unable to add another row: all rows have already been defined.");
667 }
668
669 // Add the row pattern
670 m_pattern.pushBack(nRowNZ, rowPattern);
671
672 // Add row values
673 m_values.insert(m_values.end(), rowValues, rowValues + nRowNZ);
674
675 // Update the non-zero element counters
676 m_maxRowNZ = std::max(nRowNZ, m_maxRowNZ);
677 m_nNZ += nRowNZ;
678
679 // Update the index of the last row
680 m_lastRow++;
681}

751std::unique_ptr<SparseMatrix> SparseMatrix::computeTranspose() const
752{
753 // Early return if the matrix is not assembled.
754 if (!isAssembled()) {
755 return nullptr;
756 }
757
758 // Create the transpose matrix
759 std::unique_ptr<SparseMatrix> transpose;
760#if BITPIT_ENABLE_MPI==1
761 transpose = std::unique_ptr<SparseMatrix>(new SparseMatrix(m_communicator, m_partitioned, m_nCols, m_nRows, m_nNZ));
762#else
763 transpose = std::unique_ptr<SparseMatrix>(new SparseMatrix(m_nCols, m_nRows, m_nNZ));
764#endif
765
766 // Create an empty transpose pattern
767 std::vector<std::size_t> transposeRowSizes(transpose->m_nRows, 0.);
768 for (int i = 0; i < m_nNZ; ++i) {
769 long column = *(m_pattern.data() + i);
770 ++transposeRowSizes[column];
771 }
772
773 transpose->m_pattern.initialize(transpose->m_nRows, transposeRowSizes.data(), 0.);
774
775 // Create the empty storage for the values
776 transpose->m_values.resize(m_nNZ);
777
778 // Set non-zero information
779 transpose->m_nNZ = m_nNZ;
780
781 transpose->m_maxRowNZ = 0;
782 for (int i = 0; i < transpose->m_nRows; ++i) {
783 transpose->m_maxRowNZ = std::max(static_cast<long>(transposeRowSizes[i]), transpose->m_maxRowNZ);
784 }
785
786 // Fill patter and values of the transpose matrix
787 const std::size_t *rowExtents = m_pattern.indices();
788 std::fill(transposeRowSizes.begin(), transposeRowSizes.end(), 0);
789 for (long row = 0; row < m_nRows; ++row) {
790 const std::size_t rowPatternSize = rowExtents[row + 1] - rowExtents[row];
791 const long *rowPattern = m_pattern.data() + rowExtents[row];
792 const double *rowValues = m_values.data() + rowExtents[row];
793 for (std::size_t k = 0; k < rowPatternSize; ++k) {
794 long column = rowPattern[k];
795 const std::size_t transposeRowLastIndex = *(transpose->m_pattern.indices(column)) + transposeRowSizes[column];
796
797 long *transposeRowLastPattern = transpose->m_pattern.data() + transposeRowLastIndex;
798 *transposeRowLastPattern = row;
799
800 double *transposeRowLastValue = transpose->m_values.data() + transposeRowLastIndex;
801 *transposeRowLastValue = rowValues[k];
802
803 ++transposeRowSizes[column];
804 }
805 }
806 transpose->m_lastRow = transpose->m_nRows - 1;
807
808 // Assembly the transpose matrix
809 transpose->assembly();
810
811 return transpose;
812}
