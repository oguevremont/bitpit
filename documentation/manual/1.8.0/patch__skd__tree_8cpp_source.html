70void SkdPatchInfo::buildCache(const PatchKernel::CellConstRange &cellRange)
71{
72 m_cellBoxes = std::unique_ptr<BoxCache>(new BoxCache(2, &(m_patch->getCells())));
73 m_cellCentroids = std::unique_ptr<BoxCache>(new CentroidCache(1, &(m_patch->getCells())));
74 for (auto itr = cellRange.cbegin(); itr != cellRange.cend(); ++itr) {
75 std::size_t rawCellId = itr.getRawIndex();
76
77 // Cell info
78 const Cell &cell = *itr;
79 ConstProxyVector<long> cellConnect = cell.getVertexIds();
80 int nCellVertices = cellConnect.size();
81
82 // Bounding box
83 std::array<double, 3> &cellBoxMin = m_cellBoxes->rawAt(rawCellId, 0);
84 std::array<double, 3> &cellBoxMax = m_cellBoxes->rawAt(rawCellId, 1);
85 std::array<double, 3> &cellCentroid = m_cellCentroids->rawAt(rawCellId, 0);
86
87 cellBoxMin = m_patch->getVertexCoords(cellConnect[0]);
88 cellBoxMax = cellBoxMin;
89 cellCentroid = cellBoxMin;
90 for (int i = 1; i < nCellVertices; ++i) {
91 const std::array<double, 3> &coords = m_patch->getVertexCoords(cellConnect[i]);
92 for (int d = 0; d < 3; ++d) {
93 cellBoxMin[d] = std::min(coords[d], cellBoxMin[d]);
94 cellBoxMax[d] = std::max(coords[d], cellBoxMax[d]);
95 cellCentroid[d] += coords[d];
96 }
97 }
98 cellCentroid /= double(nCellVertices);
99 }
100}

407bool SkdBox::boxContainsPoint(const std::array<double, 3> &point, double offset) const
408{
409 if (isEmpty()) {
410 return false;
411 }
412
413 for (int d = 0; d < 3; d++) {
414 if (point[d] < (m_boxMin[d] - offset)) {
415 return false;
416 }
417
418 if (point[d] > (m_boxMax[d] + offset)) {
419 return false;
420 }
421 }
422
423 return true;
424}

434bool SkdBox::boxIntersectsSphere(const std::array<double, 3> &center, double radius) const
435{
436 // If the box contains the center it will also intersect the sphere
437 if (boxContainsPoint(center, 0.)) {
438 return true;
439 }
440
441 // Check if the distance between the center and its projection on
442 // the bounding box is smaller than the radius of the sphere.
443 std::array<double, 3> delta;
444 for (int d = 0; d < 3; d++) {
445 delta[d] = std::min(std::max(center[d], m_boxMin[d]), m_boxMax[d]) - center[d];
446 }
447 double distance = dotProduct(delta, delta);
448
449 return (distance < radius * radius);
450}

493void SkdNode::initializeBoundingBox()
494{
495 // Early return if there are no cells
496 if (m_cellRangeBegin == m_cellRangeEnd){
497 m_boxMin.fill( std::numeric_limits<double>::max());
498 m_boxMax.fill(- std::numeric_limits<double>::max());
499
500 return;
501 }
502
503 // Evaluate the bounding box
504 const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
505
506 m_boxMin = m_patchInfo->getCachedBoxMin(cellRawIds[m_cellRangeBegin]);
507 m_boxMax = m_patchInfo->getCachedBoxMax(cellRawIds[m_cellRangeBegin]);
508 for (std::size_t n = m_cellRangeBegin + 1; n < m_cellRangeEnd; n++) {
509 const std::size_t rawCellId = cellRawIds[n];
510 const std::array<double, 3> &cellBoxMin = m_patchInfo->getCachedBoxMin(rawCellId);
511 const std::array<double, 3> &cellBoxMax = m_patchInfo->getCachedBoxMax(rawCellId);
512 for (int d = 0; d < 3; ++d) {
513 m_boxMin[d] = std::min(cellBoxMin[d], m_boxMin[d]);
514 m_boxMax[d] = std::max(cellBoxMax[d], m_boxMax[d]);
515 }
516 }
517
518 // Inlfate the bounding box by a small epsilon
519 //
520 // The small inflation allows us to test intersection of a query
521 // against a safety box and be assured that non-intersection of
522 // with the box implies non-intersection with all the contents
523 // in the box, even when computations are susceptible to round-off
524 // error.
525 double tolerance = m_patchInfo->getPatch().getTol();
526
527 m_boxMin -= tolerance;
528 m_boxMax += tolerance;
529}

712void SkdNode::updatePointClosestCell(const std::array<double, 3> &point, bool interiorCellsOnly,
713 long *closestId, double *closestDistance) const
714{
715 if (getCellCount() == 0) {
716 return;
717 }
718
719 const PatchKernel &patch = m_patchInfo->getPatch();
720 const PiercedVector<Cell> &cells = patch.getCells();
721 const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
722
723 for (std::size_t n = m_cellRangeBegin; n < m_cellRangeEnd; n++) {
724 std::size_t cellRawId = cellRawIds[n];
725 const Cell &cell = cells.rawAt(cellRawId);
726 if (interiorCellsOnly && !cell.isInterior()) {
727 continue;
728 }
729
730
731 updatePointClosestCell(point, cell, closestId, closestDistance);
732 }
733}

752void SkdNode::updatePointClosestCell(const std::array<double, 3> &point, const Cell &cell,
753 long *closestId, double *closestDistance) const
754{
755 const PatchKernel &patch = m_patchInfo->getPatch();
756
757 // Cell id
758 long cellId = cell.getId();
759
760 // Project the point on the cell
761 int nCellVertices = cell.getVertexCount();
762 BITPIT_CREATE_WORKSPACE(cellVertexCoordinates, std::array<double BITPIT_COMMA 3>, nCellVertices, ReferenceElementInfo::MAX_ELEM_VERTICES);
763 patch.getElementVertexCoordinates(cell, cellVertexCoordinates);
764
765 double cellDistance;
766 std::array<double, 3> cellProjection;
767 cell.evalPointProjection(point, cellVertexCoordinates, &cellProjection, &cellDistance);
768
769 // Detect if the specified cell is closer than the current closest cell
770 const int DISTANCE_CLOSER = - 1;
771 const int DISTANCE_EQUAL = 0;
772 const int DISTANCE_FARTHER = 1;
773
774 int distanceFlag = DISTANCE_FARTHER;
775 if (utils::DoubleFloatingEqual()(cellDistance, *closestDistance, patch.getTol(), patch.getTol())) {
776 distanceFlag = DISTANCE_EQUAL;
777 } else if (cellDistance < *closestDistance) {
778 distanceFlag = DISTANCE_CLOSER;
779 }
780
781 // Consider the case where no closest cell is defined
782 //
783 // Even if the id of the closest cell is null, we may have an
784 // estimated of the closest cell distance. We need to update
785 // the closest cell information only if the current cells is
786 // closer than the estimate.
787 if (*closestId == Cell::NULL_ID) {
788 if (distanceFlag != DISTANCE_FARTHER) {
789 *closestId = cellId;
790 *closestDistance = cellDistance;
791 }
792
793 return;
794 }
795
796 // Update closest cell information accordingly to the distance flag:
797 // - if the specified cell is closer the specified cell will become the
798 // closest cell
799 // - if the two cells have the same distance, the closest cell will be
800 // chosen using the normal of the cells.
801 // - if the specified cell is farther there is nothing to do;
802 switch (distanceFlag) {
803
804 case DISTANCE_CLOSER:
805 {
806 *closestId = cellId;
807 *closestDistance = cellDistance;
808
809 break;
810 }
811
812 case DISTANCE_EQUAL:
813 {
814 // Project point ont the closest cell
815 const Cell &closest = patch.getCell(*closestId);
816
817 int nClosestCellVertices = closest.getVertexCount();
818 BITPIT_CREATE_WORKSPACE(closestCellVertexCoordinates, std::array<double BITPIT_COMMA 3>, nClosestCellVertices, ReferenceElementInfo::MAX_ELEM_VERTICES);
819 patch.getElementVertexCoordinates(closest, closestCellVertexCoordinates);
820
821 double closestCellDistance;
822 std::array<double, 3> closestCellProjection;
823 closest.evalPointProjection(point, closestCellVertexCoordinates, &closestCellProjection, &closestCellDistance);
824
825 // Find normal of the cells at the centroids
826 //
827 // To be more precise, the normals should beevaluated on projection
828 // points.
829 std::array<double, 3> cellNormal = cell.evalNormal(cellVertexCoordinates);
830 std::array<double, 3> closestNormal = closest.evalNormal(closestCellVertexCoordinates);
831
832 // Find cell more aligned with respect to the point
833 //
834 // We want to chhose the cell whose normal is most aligned towards
835 // the point.
836 double cellAligement = dotProduct(cellNormal, point - cellProjection);
837 double closestAligement = dotProduct(closestNormal, point - closestCellProjection);
838
839 // If the specified cell is more aligned than the closest cell, update
840 // the closest info.
841 if (cellAligement > closestAligement) {
842 *closestId = cellId;
843 *closestDistance = cellDistance;
844 }
845
846 break;
847 }
848
849 default:
850 {
851 // Nothing to do
852 }
853
854 }
855}

913void SkdGlobalCellDistance::executeMPIMinOperation(SkdGlobalCellDistance *in, SkdGlobalCellDistance *inout, int *len, MPI_Datatype *datatype)
914{
915 BITPIT_UNUSED(datatype);
916
917 int nDistances = *len;
918 for (int i = 0; i < nDistances; ++i) {
919 bool updateDistance = false;
920 if (in[i].m_id != Cell::NULL_ID) {
921 if (inout[i].m_id == Cell::NULL_ID) {
922 updateDistance = true;
923 } else if (std::abs(in[i].m_distance) < std::abs(inout[i].m_distance)) {
924 updateDistance = true;
925 }
926 }
927
928 if (updateDistance) {
929 inout[i] = in[i];
930 }
931 }

1092void PatchSkdTree::build(std::size_t leafThreshold, bool squeezeStorage)
1093{
1094 const PatchKernel &patch = m_patchInfo.getPatch();
1095
1096 // Clear existing tree
1097 clear();
1098
1099 // Initialize list of cell raw ids
1100 std::size_t nCells;
1101 PatchKernel::CellConstRange cellRange;
1102 if (m_interiorCellsOnly) {
1103 nCells = patch.getInternalCellCount();
1104 cellRange.initialize(patch.internalCellConstBegin(), patch.internalCellConstEnd());
1105 } else {
1106 nCells = patch.getCellCount();
1107 cellRange.initialize(patch.cellConstBegin(), patch.cellConstEnd());
1108 }
1109
1110 if (m_cellRawIds.size() != nCells) {
1111 m_cellRawIds.resize(nCells);
1112 }
1113
1114 std::size_t k = 0;
1115 for (auto itr = cellRange.begin(); itr != cellRange.end(); ++itr) {
1116 m_cellRawIds[k++] = itr.getRawIndex();
1117 }
1118
1119 // Build patch cache
1120 m_patchInfo.buildCache(cellRange);
1121
1122 // Initialize node list
1123 std::size_t nodesCount = std::max(1, int(std::ceil(2. * nCells / leafThreshold - 1.)));
1124 m_nodes.reserve(nodesCount);
1125
1126 // Create the root
1127 m_nodes.emplace_back(&m_patchInfo, 0, nCells);
1128
1129 // Create the tree
1130 std::queue<std::size_t> nodeStack;
1131 nodeStack.push(0);
1132 while (!nodeStack.empty()) {
1133 std::size_t nodeId = nodeStack.front();
1134 nodeStack.pop();
1135
1136 // Create the children of the node
1137 //
1138 // This function may add new nodes to the tree, this may invalidate
1139 // any reference or pointer to the nodes. To avoid potential problems
1140 // we pass to the function a node id.
1141 createChildren(nodeId, leafThreshold);
1142
1143 // Add the newly created childrend to the stack
1144 const SkdNode &node = getNode(nodeId);
1145 for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
1146 std::size_t childId = node.getChildId(static_cast<SkdNode::ChildLocation>(i));
1147 if (childId != SkdNode::NULL_ID) {
1148 nodeStack.push(childId);
1149 }
1150 }
1151 }
1152
1153 // Squeeze storage
1154 if (squeezeStorage) {

1289void PatchSkdTree::createChildren(std::size_t parentId, std::size_t leafThreshold)
1290{
1291 const SkdNode &parent = getNode(parentId);
1292
1293 // Check if the parent is a leaf
1294 std::size_t parentCellCount = parent.getCellCount();
1295 if (parentCellCount <= leafThreshold) {
1296 createLeaf(parentId);
1297 return;
1298 }
1299
1300 // Evaluate the preferred direction along which elements will be split.
1301 //
1302 // The elements will be split along a plane normal to the direction
1303 // for which the bounding box has the maximum length.
1304 const std::array<double, 3> &parentBoxMin = parent.getBoxMin();
1305 const std::array<double, 3> &parentBoxMax = parent.getBoxMax();
1306
1307 int largerDirection = 0;
1308 double boxMaximumLength = parentBoxMax[largerDirection] - parentBoxMin[largerDirection];
1309 for (int d = 1; d < 3; ++d) {
1310 double length = parentBoxMax[d] - parentBoxMin[d];
1311 if (length > boxMaximumLength) {
1312 largerDirection = d;
1313 boxMaximumLength = length;
1314 }
1315 }
1316
1317 // Split the elements.
1318 //
1319 // Preferred direction is tried first, if the split along this direction
1320 // will produce and empty child (e.g., all the cell's box centroids have
1321 // the same coordinate along the split direction and therefore all the
1322 // cells would be assigned to the left chiled) the split will be performed
1323 // along one of the other directions.
1324 std::array<double, 3> parentWeightedCentroid = parent.evalBoxWeightedMean();
1325 for (int d = 0; d < 3; ++d) {
1326 // Update the split direction
1327 int splitDirection = (largerDirection + d) % 3;
1328
1329 // Get the threshold for the split
1330 double splitThreshold = parentWeightedCentroid[splitDirection];
1331
1332 // Order the elements
1333 //
1334 // All the elements with a centroid coordinate less or equal than the
1335 // threshold will be assigned to the left child, the others will be
1336 // assigned to the right child.
1337 std::size_t leftBegin = parent.m_cellRangeBegin;
1338 std::size_t leftEnd = parent.m_cellRangeEnd;
1339 std::size_t rightBegin = parent.m_cellRangeBegin;
1340 std::size_t rightEnd = parent.m_cellRangeEnd;
1341 while (true) {
1342 // Update the right begin
1343 while (rightBegin != leftEnd && m_patchInfo.getCachedCentroid(m_cellRawIds[rightBegin])[splitDirection] <= splitThreshold) {
1344 rightBegin++;
1345 }
1346
1347 // Update the left end
1348 while (rightBegin != leftEnd && m_patchInfo.getCachedCentroid(m_cellRawIds[leftEnd - 1])[splitDirection] > splitThreshold) {
1349 leftEnd--;
1350 }
1351
1352 // If all the elements are in the right position we can exit
1353 if (rightBegin == leftEnd) {
1354 break;
1355 }
1356
1357 // If left end and and right begin are not equal, that the two ids
1358 // point to misplaced elements. Swap the elements, advance the ids
1359 // and continue iterating.
1360 std::iter_swap(m_cellRawIds.begin() + (leftEnd - 1), m_cellRawIds.begin() + rightBegin);
1361 }
1362
1363 if (leftEnd <= leftBegin || rightEnd <= rightBegin) {
1364 continue;
1365 }
1366
1367 // Create the left child
1368 //
1369 // Adding new nodes may invalidate any pointer and reference to the
1370 // nodes, we cannot store a reference to the parent node.
1371 long leftId = m_nodes.size();
1372 m_nodes.emplace_back(&m_patchInfo, leftBegin, leftEnd);
1373 _getNode(parentId).m_children[static_cast<std::size_t>(SkdNode::CHILD_LEFT)] = leftId;
1374
1375 // Create the right child
1376 //
1377 // Adding new nodes may invalidate any pointer and reference to the
1378 // nodes, we cannot store a reference to the parent node.
1379 long rightId = m_nodes.size();
1380 m_nodes.emplace_back(&m_patchInfo, rightBegin, rightEnd);
1381 _getNode(parentId).m_children[static_cast<std::size_t>(SkdNode::CHILD_RIGHT)] = rightId;
1382
1383 // Done
1384 return;
1385 }
1386
1387 // It was not possible to split the elements. They have all the same
1388 // characteristic position and therefore they will be clustered together
1389 // in a leaf node.
1390 createLeaf(parentId);
1391}

1504void PatchSkdTree::buildPartitionBoxes()
1505{
1506 // Recover local bounding box of the root node
1507 const SkdBox &box = getNode(0).getBoundingBox();
1508
1509 // Collect minimum and maximum coordinates of bounding boxes
1510 std::vector<int> count(m_nProcessors, 3);
1511
1512 std::vector<int> displacements(m_nProcessors);
1513 for (int i = 0; i < m_nProcessors; ++i) {
1514 displacements[i] = i*3;
1515 }
1516
1517 std::vector<std::array<double, 3>> minBoxes(m_nProcessors);
1518 minBoxes[m_rank] = box.getBoxMin();
1519 MPI_Allgatherv(MPI_IN_PLACE, 3, MPI_DOUBLE, minBoxes.data(),
1520 count.data(), displacements.data(), MPI_DOUBLE, m_communicator);
1521
1522 std::vector<std::array<double, 3>> maxBoxes(m_nProcessors);
1523 maxBoxes[m_rank] = box.getBoxMax();
1524 MPI_Allgatherv(MPI_IN_PLACE, 3, MPI_DOUBLE, maxBoxes.data(),
1525 count.data(), displacements.data(), MPI_DOUBLE, m_communicator);
1526
1527 // Fill partition boxes
1528 m_partitionBoxes.resize(getPatch().getProcessorCount());
1529 for (int i = 0; i < m_nProcessors; ++i) {
1530 m_partitionBoxes[i] = SkdBox(minBoxes[i], maxBoxes[i]);
1531 }
1532}
