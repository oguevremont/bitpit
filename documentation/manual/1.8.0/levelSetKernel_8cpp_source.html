97void LevelSetKernel::updateGeometryCache( const std::vector<adaption::Info> &adaptionData ) {
98
99 // If there are no cells in the mesh we can just delete all the cache
100 if ( m_mesh->getCellCount() == 0) {
101 clearGeometryCache();
102 return;
103 }
104
105 // Remove the previous cells from the cache
106 for ( const adaption::Info &adaptionInfo : adaptionData ){
107 if( adaptionInfo.entity != adaption::Entity::ENTITY_CELL ){
108 continue;
109 }
110
111 for ( auto & previousId : adaptionInfo.previous){
112 auto centroidItr = m_cellCentroids.find( previousId ) ;
113 if ( centroidItr == m_cellCentroids.end() ) {
114 continue ;
115 }
116
117 m_cellCentroids.erase( previousId ) ;
118 }
119 }
120
121}

132const std::array<double,3> & LevelSetKernel::computeCellCentroid( long id ) const {
133
134 auto centroidItr = m_cellCentroids.find( id ) ;
135 if ( centroidItr == m_cellCentroids.end() ) {
136 centroidItr = m_cellCentroids.insert( { id, m_mesh->evalCellCentroid( id ) } ).first ;
137 }
138
139 return centroidItr->second;
140

148double LevelSetKernel::computeCellIncircle( long id ) const {
149
150 VolumeKernel *patch = getMesh();
151 Cell &cell = patch->getCell(id);
152
153 const long* interfaceIds = cell.getInterfaces();
154 int interfaceCount = cell.getInterfaceCount();
155
156 const std::array<double,3> &cellCenter = computeCellCentroid(id);
157
158 double radius = std::numeric_limits<double>::max() ;
159 for (int k = 0; k < interfaceCount; ++k) {
160 long interfaceId = interfaceIds[k];
161 double r = norm2(cellCenter - patch->evalInterfaceCentroid(interfaceId));
162 radius = std::min(radius, r);
163 }
164
165 ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
166 int nCellVertices = cellVertexIds.size();
167 for (int k = 0; k < nCellVertices; ++k) {
168 long vertexId = cellVertexIds[k];
169 double r = norm2(cellCenter - patch->getVertexCoords(vertexId));
170 radius = std::min(radius, r);
171 }
172
173 return radius;
174
175}

229double LevelSetKernel::isCellInsideBoundingBox( long id, const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax ) const {
230
231 double distanceTolerance = m_mesh->getTol();
232
233 std::array<double,3> cellBoxMin;
234 std::array<double,3> cellBoxMax;
235 m_mesh->evalCellBoundingBox(id, &cellBoxMin, &cellBoxMax);

245void LevelSetKernel::initializeCommunicator()
246{
247 // The communicator has to be valid
248 MPI_Comm communicator = m_mesh->getCommunicator();
249 if (communicator == MPI_COMM_NULL) {
250 throw std::runtime_error ("Levelset communicator is not valid");
251 }
252
253 // Communication can be set just once
254 if (isCommunicatorSet()) {
255 int comparison;
256 MPI_Comm_compare(communicator, m_communicator, &comparison);
257 if (comparison == MPI_IDENT || comparison == MPI_CONGRUENT) {
258 return;
259 } else {
260 throw std::runtime_error ("Levelset communicator cannot be modified.");
261 }
262 }
263
264 // Create a duplicate of the patch communicator
265 MPI_Comm_dup(communicator, &m_communicator);
266}
