214std::array<double, 3> VolUnstructured::evalInterfaceNormal(long id) const
215{
216 const Interface &interface = getInterface(id);
217
218 ConstProxyVector<long> interfaceVertexIds = interface.getVertexIds();
219 std::size_t nInterfaceVertices = interfaceVertexIds.size();
220 BITPIT_CREATE_WORKSPACE(vertexCoordinates, std::array<double BITPIT_COMMA 3>, nInterfaceVertices, ReferenceElementInfo::MAX_ELEM_VERTICES);
221 getVertexCoords(nInterfaceVertices, interfaceVertexIds.data(), vertexCoordinates);
222
223 std::array<double, 3> orientation = {{0., 0., 0.}};
224 if (!isThreeDimensional()) {
225 long ownerId = interface.getOwner();
226 const Cell &owner = getCell(ownerId);
227
228 ConstProxyVector<long> ownerVertexIds = owner.getVertexIds();
229 const int nOwnerVertices = ownerVertexIds.size();
230
231 const std::array<double, 3> &V_A = getVertex(ownerVertexIds[0]).getCoords();
232 const std::array<double, 3> &V_B = getVertex(ownerVertexIds[1]).getCoords();
233 const std::array<double, 3> &V_Z = getVertex(ownerVertexIds[nOwnerVertices - 1]).getCoords();
234
235 orientation = crossProduct(V_B - V_A, V_Z - V_A);
236 }
237
238 return interface.evalNormal(vertexCoordinates, orientation);
239}
